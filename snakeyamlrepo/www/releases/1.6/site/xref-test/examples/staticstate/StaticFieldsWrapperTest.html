32  /***
33   * Example: using wrapper object for static fields
34   */
35  public class StaticFieldsWrapperTest extends TestCase {
36  
37      /***
38       * use wrapper with global tag
39       */
40      public void testWrapper() {
41          JavaBeanWithStaticState bean = new JavaBeanWithStaticState();
42          bean.setName("Bahrack");
43          bean.setAge(-47);
44          JavaBeanWithStaticState.setType("Type3");
45          JavaBeanWithStaticState.color = "Violet";
46          Yaml yaml = new Yaml();
47          String output = yaml.dump(new Wrapper(bean));
48          // System.out.println(output);
49          assertEquals(
50                  "!!examples.staticstate.Wrapper {age: -47, color: Violet, name: Bahrack, type: Type3}\n",
51                  output);
52          // parse back to instance
53          Wrapper wrapper = (Wrapper) yaml.load(output);
54          JavaBeanWithStaticState bean2 = wrapper.createBean();
55          assertEquals(-47, bean2.getAge());
56          assertEquals("Bahrack", bean2.getName());
57      }
58  
59      /***
60       * use wrapper with local tag
61       */
62      public void testLocalTag() {
63          JavaBeanWithStaticState bean = new JavaBeanWithStaticState();
64          bean.setName("Bahrack");
65          bean.setAge(-47);
66          JavaBeanWithStaticState.setType("Type3");
67          JavaBeanWithStaticState.color = "Violet";
68          Representer repr = new Representer();
69          repr.addClassTag(Wrapper.class, new Tag("!mybean"));
70          Yaml yaml = new Yaml(new Dumper(repr, new DumperOptions()));
71          String output = yaml.dump(new Wrapper(bean));
72          // System.out.println(output);
73          assertEquals("!mybean {age: -47, color: Violet, name: Bahrack, type: Type3}\n", output);
74          // parse back to instance
75          Constructor constr = new Constructor();
76          TypeDescription description = new TypeDescription(Wrapper.class, new Tag("!mybean"));
77          constr.addTypeDescription(description);
78          yaml = new Yaml(new Loader(constr));
79          Wrapper wrapper = (Wrapper) yaml.load(output);
80          JavaBeanWithStaticState bean2 = wrapper.createBean();
81          assertEquals(-47, bean2.getAge());
82          assertEquals("Bahrack", bean2.getName());
83      }
84  
85      /***
86       * use wrapper with no tag
87       */
88      public void testRootBean() {
89          JavaBeanWithStaticState bean = new JavaBeanWithStaticState();
90          bean.setName("Bahrack");
91          bean.setAge(-47);
92          JavaBeanWithStaticState.setType("Type3");
93          JavaBeanWithStaticState.color = "Violet";
94          DumperOptions options = new DumperOptions();
95          options.setDefaultFlowStyle(FlowStyle.FLOW);
96          options.setExplicitRoot(Tag.MAP);
97          JavaBeanDumper dumper = new JavaBeanDumper(new Representer(), options);
98          String output = dumper.dump(new Wrapper(bean));
99          // System.out.println(output);
100         assertEquals("{age: -47, color: Violet, name: Bahrack, type: Type3}\n", output);
101         // parse back to instance
102         JavaBeanLoader<Wrapper> loader = new JavaBeanLoader<Wrapper>(Wrapper.class);
103         Wrapper wrapper = loader.load(output);
104         JavaBeanWithStaticState bean2 = wrapper.createBean();
105         assertEquals(-47, bean2.getAge());
106         assertEquals("Bahrack", bean2.getName());
107     }
