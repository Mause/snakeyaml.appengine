16  package org.yaml.snakeyaml.types;
17  
18  import java.io.IOException;
19  import java.util.HashMap;
20  import java.util.Map;
21  
22  /***
23   * @see http://yaml.org/type/int.html
24   */
25  public class BoolTagTest extends AbstractTest {
26      public void testBool() throws IOException {
27          assertEquals(Boolean.TRUE, getMapValue("canonical: true", "canonical"));
28          assertEquals(Boolean.FALSE, getMapValue("answer: NO", "answer"));
29          assertEquals(Boolean.TRUE, getMapValue("logical: True", "logical"));
30          assertEquals(Boolean.TRUE, getMapValue("option: on", "option"));
31      }
32  
33      public void testBoolCanonical() throws IOException {
34          assertEquals(Boolean.TRUE, getMapValue("canonical: Yes", "canonical"));
35          assertEquals(Boolean.TRUE, getMapValue("canonical: yes", "canonical"));
36          assertEquals(Boolean.TRUE, getMapValue("canonical: YES", "canonical"));
37          assertEquals("yES", getMapValue("canonical: yES", "canonical"));
38          assertEquals(Boolean.FALSE, getMapValue("canonical: No", "canonical"));
39          assertEquals(Boolean.FALSE, getMapValue("canonical: NO", "canonical"));
40          assertEquals(Boolean.FALSE, getMapValue("canonical: no", "canonical"));
41          assertEquals(Boolean.FALSE, getMapValue("canonical: off", "canonical"));
42          assertEquals(Boolean.FALSE, getMapValue("canonical: Off", "canonical"));
43          assertEquals(Boolean.FALSE, getMapValue("canonical: OFF", "canonical"));
44          assertEquals(Boolean.TRUE, getMapValue("canonical: ON", "canonical"));
45          assertEquals(Boolean.TRUE, getMapValue("canonical: On", "canonical"));
46          assertEquals(Boolean.TRUE, getMapValue("canonical: on", "canonical"));
47          // it looks like it is against the specification but it is like in
48          // PyYAML
49          assertEquals("n", getMapValue("canonical: n", "canonical"));
50          assertEquals("N", getMapValue("canonical: N", "canonical"));
51          assertEquals("y", getMapValue("canonical: y", "canonical"));
52          assertEquals("Y", getMapValue("canonical: Y", "canonical"));
53      }
54  
55      public void testBoolShorthand() throws IOException {
56          assertEquals(Boolean.TRUE, getMapValue("boolean: !!bool true", "boolean"));
57      }
58  
59      public void testBoolTag() throws IOException {
60          assertEquals(Boolean.TRUE,
61                  getMapValue("boolean: !<tag:yaml.org,2002:bool> true", "boolean"));
62      }
63  
64      public void testBoolOut() throws IOException {
65          Map<String, Boolean> map = new HashMap<String, Boolean>();
66          map.put("boolean", Boolean.TRUE);
67          String output = dump(map);
68          assertTrue(output, output.contains("boolean: true"));
69      }
