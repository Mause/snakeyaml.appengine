30  public class Representer extends SafeRepresenter {
31      private Map<Class<? extends Object>, String> classTags;
32      private Map<Class<? extends Object>, TypeDescription> classDefinitions;

34      public Representer(Character default_style, Boolean default_flow_style) {
35          super(default_style, default_flow_style);
36          classTags = new HashMap<Class<? extends Object>, String>();
37          classDefinitions = new HashMap<Class<? extends Object>, TypeDescription>();
38          this.representers.put(null, new RepresentJavaBean());
39      }

41      public Representer() {
42          this(null, null);
43      }

45      /***
46       * Make YAML aware how to represent a custom Class. If there is no root
47       * Class assigned in constructor then the 'root' property of this definition
48       * is respected.
49       * 
50       * @param definition
51       *            to be added to the Constructor
52       * @return the previous value associated with <tt>definition</tt>, or
53       *         <tt>null</tt> if there was no mapping for <tt>definition</tt>.
54       */
55      public TypeDescription addTypeDescription(TypeDescription definition) {
56          if (definition == null) {
57              throw new NullPointerException("ClassDescription is required.");
58          }
59          String tag = definition.getTag();
60          classTags.put(definition.getType(), tag);
61          return classDefinitions.put(definition.getType(), definition);
62      }

64      private class RepresentJavaBean implements Represent {
65          public Node representData(Object data) {
66              Set<Property> properties;
67              try {
68                  properties = getProperties(data.getClass());
69              } catch (IntrospectionException e) {
70                  throw new YAMLException(e);
71              }
72              Node node = representMapping(properties, data);
73              return node;
74          }
75      }

77      private Node representMapping(Set<Property> properties, Object javaBean) {
78          List<Node[]> value = new LinkedList<Node[]>();
79          String tag;
80          String customTag = classTags.get(javaBean.getClass());
81          if (customTag == null) {
82              if (rootTag == null) {
83                  tag = "tag:yaml.org,2002:" + javaBean.getClass().getName();
84              } else {
85                  tag = "tag:yaml.org,2002:map";
86              }
87          } else {
88              tag = customTag;
89          }
90          if (rootTag == null) {
91              rootTag = tag;
92          }
93          // flow style will be chosen by BaseRepresenter
94          MappingNode node = new MappingNode(tag, value, null);
95          representedObjects.put(aliasKey, node);
96          boolean bestStyle = true;
97          for (Property property : properties) {
98              Node nodeKey = representData(property.getName());
99              Object memberValue = property.get(javaBean);
100             Node nodeValue = representData(memberValue);
101             if (nodeValue instanceof MappingNode) {
102                 if (!Map.class.isAssignableFrom(memberValue.getClass())) {
103                     if (property.getType() != memberValue.getClass()) {
104                         String memberTag = "tag:yaml.org,2002:" + memberValue.getClass().getName();
105                         nodeValue.setTag(memberTag);
106                     }
107                 }
108             } else if (memberValue != null && Enum.class.isAssignableFrom(memberValue.getClass())) {
109                 nodeValue.setTag("tag:yaml.org,2002:str");
110             }
111             if (!((nodeKey instanceof ScalarNode && ((ScalarNode) nodeKey).getStyle() == null))) {
112                 bestStyle = false;
113             }
114             if (!((nodeValue instanceof ScalarNode && ((ScalarNode) nodeValue).getStyle() == null))) {
115                 bestStyle = false;
116             }
117             value.add(new Node[] { nodeKey, nodeValue });
118         }
119         if (defaultFlowStyle != null) {
120             node.setFlowStyle(defaultFlowStyle);
121         } else {
122             node.setFlowStyle(bestStyle);
123         }
124         return node;
125     }

127     private Set<Property> getProperties(Class<? extends Object> type) throws IntrospectionException {
128         if (type == null) {
129             throw new NullPointerException("type cannot be null.");
130         }
131         Set<Property> properties = new TreeSet<Property>();
132         for (PropertyDescriptor property : Introspector.getBeanInfo(type).getPropertyDescriptors())
133             if (property.getReadMethod() != null
134                     && !property.getReadMethod().getName().equals("getClass")) {
135                 properties.add(new MethodProperty(property));
136             }
137         for (Field field : type.getFields()) {
138             int modifiers = field.getModifiers();
139             if (!Modifier.isPublic(modifiers) || Modifier.isStatic(modifiers)
140                     || Modifier.isTransient(modifiers))
141                 continue;
142             properties.add(new FieldProperty(field));
143         }
144         return properties;
145     }
