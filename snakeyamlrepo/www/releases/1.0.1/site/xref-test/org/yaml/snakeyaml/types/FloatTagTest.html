15      public void testFloat() throws IOException {
16          assertEquals(new Double(6.8523015e+5), getMapValue("canonical: 6.8523015e+5", "canonical"));
17          assertEquals(new Double(6.8523015e+5), getMapValue("exponentioal: 685.230_15e+03",
18                  "exponentioal"));
19          assertEquals(new Double(6.8523015e+5), getMapValue("fixed: 685_230.15", "fixed"));
20          assertEquals(new Double(6.8523015e+5), getMapValue("sexagesimal: 190:20:30.15",
21                  "sexagesimal"));
22          assertEquals(Double.NEGATIVE_INFINITY, getMapValue("negative infinity: -.inf",
23                  "negative infinity"));
24          assertEquals(Double.NaN, getMapValue("not a number: .NaN", "not a number"));
25      }
