15  public class AnyObjectExampleTest extends TestCase {
16      @SuppressWarnings("unchecked")
17      public void testLoad() throws IOException {
18          String doc = Util.getLocalResource("examples/any-object-example.yaml");
19          System.out.println(doc);
20          Yaml yaml = new Yaml();
21          Map<String, Object> object = (Map<String, Object>) yaml.load(doc);
22          System.out.println(object);
23          assertEquals(6, object.size());
24          assertEquals("[null, null]", object.get("none").toString());
25          List list1 = (List) object.get("none");
26          assertEquals(2, list1.size());
27          for (Object object2 : list1) {
28              assertNull(object2);
29          }
30          //
31          assertEquals("[true, false, true, false]", object.get("bool").toString());
32          assertEquals(4, ((List) object.get("bool")).size());
33          //
34          assertEquals(new Integer(42), object.get("int"));
35          assertEquals(new Double(3.14159), object.get("float"));
36          //
37          assertEquals("[LITE, RES_ACID, SUS_DEXT]", object.get("list").toString());
38          List list2 = (List) object.get("list");
39          assertEquals(3, list2.size());
40          for (Object object2 : list2) {
41              assertEquals(object2.toString(), object2.toString().toUpperCase());
42          }
43          //
44          assertEquals("{hp=13, sp=5}", object.get("dict").toString());
45          Map<String, Integer> map = (Map<String, Integer>) object.get("dict");
46          assertEquals(2, map.keySet().size());
47          assertEquals(new Integer(13), map.get("hp"));
48          assertEquals(new Integer(5), map.get("sp"));
49      }
