52  public class CanonicalParser implements Parser {
53      private LinkedList<Event> events;
54      private boolean parsed;
55      private CanonicalScanner scanner;
56  
57      public CanonicalParser(String data) {
58          events = new LinkedList<Event>();
59          parsed = false;
60          scanner = new CanonicalScanner(data);
61      }
62  
63      // stream: STREAM-START document* STREAM-END
64      private void parseStream() {
65          scanner.getToken(StreamStartToken.class);
66          events.add(new StreamStartEvent(null, null));
67          while (!scanner.checkToken(StreamEndToken.class)) {
68              List<Class<? extends Token>> list = new ArrayList<Class<? extends Token>>();
69              list.add(DirectiveToken.class);
70              list.add(DocumentStartToken.class);
71              if (scanner.checkToken(list)) {
72                  parseDocument();
73              } else {
74                  throw new CanonicalException("document is expected, got " + scanner.tokens.get(0));
75              }
76          }
77          scanner.getToken(StreamEndToken.class);
78          events.add(new StreamEndEvent(null, null));
79      }
80  
81      // document: DIRECTIVE? DOCUMENT-START node
82      private void parseDocument() {
83          if (scanner.checkToken(DirectiveToken.class)) {
84              scanner.getToken(DirectiveToken.class);
85          }
86          scanner.getToken(DocumentStartToken.class);
87          events.add(new DocumentStartEvent(null, null, true, new Integer[] { 1, 1 }, null));
88          parseNode();
89          events.add(new DocumentEndEvent(null, null, true));
90      }
91  
92      // node: ALIAS | ANCHOR? TAG? (SCALAR|sequence|mapping)
93      private void parseNode() {
94          if (scanner.checkToken(AliasToken.class)) {
95              AliasToken token = (AliasToken) scanner.getToken();
96              events.add(new AliasEvent(token.getValue(), null, null));
97          } else {
98              String anchor = null;
99              if (scanner.checkToken(AnchorToken.class)) {
100                 AnchorToken token = (AnchorToken) scanner.getToken();
101                 anchor = token.getValue();
102             }
103             String tag = null;
104             if (scanner.checkToken(TagToken.class)) {
105                 TagToken token = (TagToken) scanner.getToken();
106                 tag = token.getValue()[0] + token.getValue()[1];
107             }
108             if (scanner.checkToken(ScalarToken.class)) {
109                 ScalarToken token = (ScalarToken) scanner.getToken();
110                 events.add(new ScalarEvent(anchor, tag, new ImplicitTuple(false, false), token
111                         .getValue(), null, null, null));
112             } else if (scanner.checkToken(FlowSequenceStartToken.class)) {
113                 events.add(new SequenceStartEvent(anchor, tag, false, null, null, null));
114                 parseSequence();
115             } else if (scanner.checkToken(FlowMappingStartToken.class)) {
116                 events.add(new MappingStartEvent(anchor, tag, false, null, null, null));
117                 parseMapping();
118             } else {
119                 throw new CanonicalException("SCALAR, '[', or '{' is expected, got "
120                         + scanner.tokens.get(0));
121             }
122         }
123     }
124 
125     // sequence: SEQUENCE-START (node (ENTRY node)*)? ENTRY? SEQUENCE-END
126     private void parseSequence() {
127         scanner.getToken(FlowSequenceStartToken.class);
128         if (!scanner.checkToken(FlowSequenceEndToken.class)) {
129             parseNode();
130             while (!scanner.checkToken(FlowSequenceEndToken.class)) {
131                 scanner.getToken(FlowEntryToken.class);
132                 if (!scanner.checkToken(FlowSequenceEndToken.class)) {
133                     parseNode();
134                 }
135             }
136         }
137         scanner.getToken(FlowSequenceEndToken.class);
138         events.add(new SequenceEndEvent(null, null));
139     }
140 
141     // mapping: MAPPING-START (map_entry (ENTRY map_entry)*)? ENTRY? MAPPING-END
142     private void parseMapping() {
143         scanner.getToken(FlowMappingStartToken.class);
144         if (!scanner.checkToken(FlowMappingEndToken.class)) {
145             parseMapEntry();
146             while (!scanner.checkToken(FlowMappingEndToken.class)) {
147                 scanner.getToken(FlowEntryToken.class);
148                 if (!scanner.checkToken(FlowMappingEndToken.class)) {
149                     parseMapEntry();
150                 }
151             }
152         }
153         scanner.getToken(FlowMappingEndToken.class);
154         events.add(new MappingEndEvent(null, null));
155     }
156 
157     // map_entry: KEY node VALUE node
158     private void parseMapEntry() {
159         scanner.getToken(KeyToken.class);
160         parseNode();
161         scanner.getToken(ValueToken.class);
162         parseNode();
163     }
164 
165     public void parse() {
166         parseStream();
167         parsed = true;
168     }
169 
170     public Event getEvent() {
171         if (!parsed) {
172             parse();
173         }
174         return events.removeFirst();
175     }
176 
177     public boolean checkEvent(List<Class<? extends Event>> choices) {
178         if (!parsed) {
179             parse();
180         }
181         if (!events.isEmpty()) {
182             if (choices.isEmpty()) {
183                 return true;
184             }
185             for (Class<? extends Event> class1 : choices) {
186                 if (class1.isInstance(events.peek())) {
187                     return true;
188                 }
189             }
190         }
191         return false;
192     }
193 
194     public boolean checkEvent(Class<? extends Event> choice) {
195         List<Class<? extends Event>> list = new ArrayList<Class<? extends Event>>(1);
196         list.add(choice);
197         return checkEvent(list);
198     }
199 
200     public Event peekEvent() {
201         if (!parsed) {
202             parse();
203         }
204         if (events.isEmpty()) {
205             return null;
206         } else {
207             return events.get(0);
208         }
209     }
