67      public void testChildren() throws IOException {
68          Human father = new Human();
69          father.setName("Father");
70          father.setBirthday(new Date(1000000000));
71          father.setBirthPlace("Leningrad");
72          father.setBankAccountOwner(father);
73          //
74          Human mother = new Human();
75          mother.setName("Mother");
76          mother.setBirthday(new Date(100000000000L));
77          mother.setBirthPlace("Saint-Petersburg");
78          father.setPartner(mother);
79          mother.setPartner(father);
80          mother.setBankAccountOwner(father);
81          //
82          Human son = new Human();
83          son.setName("Son");
84          son.setBirthday(new Date(310000000000L));
85          son.setBirthPlace("Munich");
86          son.setBankAccountOwner(father);
87          son.setFather(father);
88          son.setMother(mother);
89          //
90          Human daughter = new Human();
91          daughter.setName("Daughter");
92          daughter.setBirthday(new Date(420000000000L));
93          daughter.setBirthPlace("New York");
94          daughter.setBankAccountOwner(father);
95          daughter.setFather(father);
96          daughter.setMother(mother);
97          //
98          Set<Human> children = new LinkedHashSet<Human>(2);
99          children.add(son);
100         children.add(daughter);
101         father.setChildren(children);
102         mother.setChildren(children);
103         //
104         JavaBeanDumper beanDumper = new JavaBeanDumper();
105         String output = beanDumper.dump(son);
106         // System.out.println(output);
107         String etalon = Util.getLocalResource("recursive/with-children.yaml");
108         assertEquals(etalon, output);
109         TypeDescription humanDescription = new TypeDescription(Human.class);
110         humanDescription.putMapPropertyType("children", Human.class, Object.class);
111         JavaBeanLoader<Human> beanLoader = new JavaBeanLoader<Human>(humanDescription);
112         //
113         Human son2 = beanLoader.load(output);
114         assertNotNull(son2);
115         assertEquals("Son", son.getName());
116 
117         Human father2 = son2.getFather();
118         assertEquals("Father", father2.getName());
119         assertEquals("Mother", son2.getMother().getName());
120         assertSame(father2, father2.getBankAccountOwner());
121         assertSame(father2.getPartner(), son2.getMother());
122         assertSame(father2, son2.getMother().getPartner());
123 
124         Set<Human> children2 = father2.getChildren();
125         assertEquals(2, children2.size());
126         assertSame(father2.getPartner().getChildren(), children2);
127 
128         for (Object child : children2) {
129             // check if type descriptor was correct
130             assertSame(Human.class, child.getClass());
131         }
