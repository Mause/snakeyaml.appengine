32  /***
33   * Public YAML interface. Each Thread must have its own instance.
34   */
35  public class Yaml {
36      private final Dumper dumper;
37      private final Loader loader;
38      private final Resolver resolver;
39      private String name;

53      /***
54       * Create Yaml instance. It is safe to create a few instances and use them
55       * in different Threads.
56       * 
57       * @param loader
58       *            Loader to parse incoming documents
59       * @param dumper
60       *            Dumper to emit outgoing objects
61       */
62      public Yaml(Loader loader, Dumper dumper) {
63          this(loader, dumper, new Resolver());
64      }

66      /***
67       * Create Yaml instance. It is safe to create a few instances and use them
68       * in different Threads.
69       * 
70       * @param loader
71       *            Loader to parse incoming documents
72       * @param dumper
73       *            Dumper to emit outgoing objects
74       * @param resolver
75       *            Resolver to detect implicit type
76       */
77      public Yaml(Loader loader, Dumper dumper, Resolver resolver) {
78          this.loader = loader;
79          loader.setAttached();
80          this.dumper = dumper;
81          dumper.setAttached();
82          this.resolver = resolver;
83          this.loader.setResolver(resolver);
84          this.name = "Yaml:" + System.identityHashCode(this);
85      }
86  
87      public Yaml() {
88          this(new Loader(), new Dumper(new DumperOptions()));
89      }

91      /***
92       * Serialize a Java object into a YAML String.
93       * 
94       * @param data
95       *            Java object to be Serialized to YAML
96       * @return YAML String
97       */
98      public String dump(Object data) {
99          List<Object> list = new ArrayList<Object>(1);
100         list.add(data);
101         return dumpAll(list.iterator());
102     }
103 
104     /***
105      * Serialize a sequence of Java objects into a YAML String.
106      * 
107      * @param data
108      *            Iterator with Objects
109      * @return YAML String with all the objects in proper sequence
110      */
111     public String dumpAll(Iterator<? extends Object> data) {
112         StringWriter buffer = new StringWriter();
113         dumpAll(data, buffer);
114         return buffer.toString();
115     }

117     /***
118      * Serialize a Java object into a YAML stream.
119      * 
120      * @param data
121      *            Java object to be Serialized to YAML
122      * @param output
123      *            stream to write to
124      */
125     public void dump(Object data, Writer output) {
126         List<Object> list = new ArrayList<Object>(1);
127         list.add(data);
128         dumpAll(list.iterator(), output);

131     /***
