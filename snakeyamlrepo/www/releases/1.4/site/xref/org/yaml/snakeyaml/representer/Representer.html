43  public class Representer extends SafeRepresenter {
44      private Map<Class<? extends Object>, String> classTags;
45  
46      public Representer() {
47          classTags = new HashMap<Class<? extends Object>, String>();
48          this.representers.put(null, new RepresentJavaBean());
49      }
50  
51      /***
52       * Define a tag for the <code>Class</code> to serialize
53       * 
54       * @param clazz
55       *            <code>Class</code> which tag is changed
56       * @param tag
57       *            new tag to be used for every instance of the specified
58       *            <code>Class</code>
59       * @return the previous tag associated with the <code>Class</code>
60       */
61      public String addClassTag(Class<? extends Object> clazz, String tag) {
62          if (tag == null) {
63              throw new NullPointerException("Tag must be provided.");
64          }
65          return classTags.put(clazz, tag);
66      }

68      private class RepresentJavaBean implements Represent {
69          public Node representData(Object data) {
70              Set<Property> properties;
71              try {
72                  properties = getProperties(data.getClass());
73              } catch (IntrospectionException e) {
74                  throw new YAMLException(e);
75              }
76              Node node = representJavaBean(properties, data);
77              return node;
78          }
79      }
80  
81      /***
82       * Tag logic:<br/>
83       * - explicit root tag is set in serializer <br/>
84       * - if there is a predefined class tag it is used<br/>
85       * - a global tag with class name is always used as tag. The JavaBean parent
86       * of the specified JavaBean may set another tag (tag:yaml.org,2002:map)
87       * when the property class is the same as runtime class
88       * 
89       * @param properties
90       *            JavaBean getters
91       * @param javaBean
92       *            instance for Node
93       * @return Node to get serialized
94       */
95      private Node representJavaBean(Set<Property> properties, Object javaBean) {
96          List<NodeTuple> value = new LinkedList<NodeTuple>();
97          String tag;
98          String customTag = classTags.get(javaBean.getClass());
99          tag = customTag != null ? customTag : Tags.getGlobalTagForClass(javaBean.getClass());
100         // flow style will be chosen by BaseRepresenter
101         MappingNode node = new MappingNode(tag, value, null);
102         representedObjects.put(objectToRepresent, node);
103         boolean bestStyle = true;
104         for (Property property : properties) {
105             ScalarNode nodeKey = (ScalarNode) representData(property.getName());
106             Object memberValue = property.get(javaBean);
107             boolean hasAlias = false;
108             if (this.representedObjects.containsKey(memberValue)) {
109                 // the first occurrence of the node must keep the tag
110                 hasAlias = true;
111             }
112             Node nodeValue = representData(memberValue);
113             // if possible try to avoid a global tag with a class name
114             if (nodeValue instanceof MappingNode && !hasAlias) {
115                 // the node is a map, set or JavaBean
116                 if (!Map.class.isAssignableFrom(memberValue.getClass())) {
117                     // the node is set or JavaBean
118                     if (property.getType() == memberValue.getClass()) {
119                         // we do not need global tag because the property
120                         // Class is the same as the runtime class
121                         nodeValue.setTag(Tags.MAP);
122                     }
123                 }
124             } else if (memberValue != null && Enum.class.isAssignableFrom(memberValue.getClass())) {
125                 nodeValue.setTag(Tags.STR);
126             }
127             if (nodeKey.getStyle() != null) {
128                 bestStyle = false;
129             }
130             if (!((nodeValue instanceof ScalarNode && ((ScalarNode) nodeValue).getStyle() == null))) {
131                 bestStyle = false;
132             }
133             value.add(new NodeTuple(nodeKey, nodeValue));
134         }
135         if (defaultFlowStyle != null) {
136             node.setFlowStyle(defaultFlowStyle);
137         } else {
138             node.setFlowStyle(bestStyle);
139         }
140         return node;
141     }

143     private Set<Property> getProperties(Class<? extends Object> type) throws IntrospectionException {
144         Set<Property> properties = new TreeSet<Property>();
145         // add JavaBean getters
146         for (PropertyDescriptor property : Introspector.getBeanInfo(type).getPropertyDescriptors())
147             if (property.getReadMethod() != null
148                     && !property.getReadMethod().getName().equals("getClass")) {
149                 properties.add(new MethodProperty(property));
150             }
151         // add public fields
152         for (Field field : type.getFields()) {
153             int modifiers = field.getModifiers();
154             if (Modifier.isStatic(modifiers) || Modifier.isTransient(modifiers))
155                 continue;
156             properties.add(new FieldProperty(field));
157         }
158         if (properties.isEmpty()) {
159             throw new YAMLException("No JavaBean properties found in " + type.getName());
160         }
161         return properties;
162     }
