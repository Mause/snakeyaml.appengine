63      /***
64       * Create Constructor for the specified class as the root.
65       * 
66       * @param theRoot
67       *            - the class (usually JavaBean) to be constructed
68       */
69      public Constructor(Class<? extends Object> theRoot) {
70          if (theRoot == null) {
71              throw new NullPointerException("Root type must be provided.");
72          }
73          this.yamlConstructors.put(null, new ConstructYamlObject());
74          if (!Object.class.equals(theRoot)) {
75              rootTag = Tags.getGlobalTagForClass(theRoot);
76          }
77          typeTags = new HashMap<String, Class<? extends Object>>();
78          typeDefinitions = new HashMap<Class<? extends Object>, TypeDescription>();
79          yamlClassConstructors.put(NodeId.scalar, new ConstructScalar());
80          yamlClassConstructors.put(NodeId.mapping, new ConstructMapping());
81          yamlClassConstructors.put(NodeId.sequence, new ConstructSequence());
82      }

84      /***
85       * Create Constructor for a class which does not have to be in the classpath
86       * or for a definition from a Spring ApplicationContext.
87       * 
88       * @param theRoot
89       *            fully qualified class name of the root class (usually
90       *            JavaBean)
91       * @throws ClassNotFoundException
92       */
93      public Constructor(String theRoot) throws ClassNotFoundException {
94          this(Class.forName(check(theRoot)));
95      }

97      private static final String check(String s) {
98          if (s == null) {
99              throw new NullPointerException("Root type must be provided.");
100         }
101         if (s.trim().length() == 0) {
102             throw new YAMLException("Root type must be provided.");
103         }
104         return s;
105     }

107     /***
108      * Make YAML aware how to parse a custom Class. If there is no root Class
109      * assigned in constructor then the 'root' property of this definition is
110      * respected.
111      * 
112      * @param definition
113      *            to be added to the Constructor
114      * @return the previous value associated with <tt>definition</tt>, or
115      *         <tt>null</tt> if there was no mapping for <tt>definition</tt>.
116      */
117     public TypeDescription addTypeDescription(TypeDescription definition) {
118         if (definition == null) {
119             throw new NullPointerException("TypeDescription is required.");
120         }
121         if (rootTag == null && definition.isRoot()) {
122             rootTag = Tags.getGlobalTagForClass(definition.getType());
123         }
124         String tag = definition.getTag();
125         typeTags.put(tag, definition.getType());
126         return typeDefinitions.put(definition.getType(), definition);
127     }

134         private final Map<Class<? extends Object>, Map<String, Property>> propertiesCache = new HashMap<Class<? extends Object>, Map<String, Property>>();

136         /***
137          * Construct JavaBean. If type safe collections are used please look at
138          * <code>TypeDescription</code>.
139          * 
140          * @param node
141          *            node where the keys are property names (they can only be
142          *            <code>String</code>s) and values are objects to be created
143          * @return constructed JavaBean
144          */
145         public Object construct(Node node) {
146             MappingNode mnode = (MappingNode) node;
147             if (Properties.class.isAssignableFrom(node.getType())) {
148                 Properties properties = new Properties();
149                 if (!node.isTwoStepsConstruction()) {
150                     constructMapping2ndStep(mnode, (Map<Object, Object>) properties);
151                 } else {
152                     throw new YAMLException("Properties must not be recursive.");
153                 }
154                 return properties;
155             } else if (SortedMap.class.isAssignableFrom(node.getType())) {
156                 SortedMap<Object, Object> map = new TreeMap<Object, Object>();
157                 if (!node.isTwoStepsConstruction()) {
158                     constructMapping2ndStep(mnode, map);
159                 }
160                 return map;
161             } else if (Map.class.isAssignableFrom(node.getType())) {
162                 if (node.isTwoStepsConstruction()) {
163                     return createDefaultMap();
164                 } else {
165                     return constructMapping(mnode);
166                 }
167             } else if (SortedSet.class.isAssignableFrom(node.getType())) {
168                 SortedSet<Object> set = new TreeSet<Object>();
169                 if (!node.isTwoStepsConstruction()) {
170                     constructSet2ndStep(mnode, set);
171                 }
172                 return set;
173             } else if (Set.class.isAssignableFrom(node.getType())) {
174                 if (node.isTwoStepsConstruction()) {
175                     return createDefaultSet();
176                 } else {
177                     return constructSet(mnode);
178                 }
179             } else {
180                 if (node.isTwoStepsConstruction()) {
181                     return createEmptyJavaBean(mnode);
182                 } else {
183                     return constructJavaBean2ndStep(mnode, createEmptyJavaBean(mnode));
184                 }
185             }
186         }

188         @SuppressWarnings("unchecked")
189         public void construct2ndStep(Node node, Object object) {
190             if (Map.class.isAssignableFrom(node.getType())) {
191                 constructMapping2ndStep((MappingNode) node, (Map<Object, Object>) object);
192             } else if (Set.class.isAssignableFrom(node.getType())) {
193                 constructSet2ndStep((MappingNode) node, (Set<Object>) object);
194             } else {
195                 constructJavaBean2ndStep((MappingNode) node, object);
196             }
197         }

199         private Object createEmptyJavaBean(MappingNode node) {
200             try {
201                 Class<? extends Object> type = node.getType();
202                 if (Modifier.isAbstract(type.getModifiers())) {
203                     node.setType(getClassForNode(node));
204                 }
205                 /***
206                  * Using only default constructor. Everything else will be
207                  * initialized on 2nd step. If we do here some partial
208                  * initialization, how do we then track what need to be done on
209                  * 2nd step? I think it is better to get only object here (to
210                  * have it as reference for recursion) and do all other thing on
211                  * 2nd step.
212                  */
213                 return node.getType().newInstance();
214             } catch (InstantiationException e) {
215                 throw new YAMLException(e);
216             } catch (IllegalAccessException e) {
217                 throw new YAMLException(e);
218             }
219         }

221         @SuppressWarnings("unchecked")
222         private Object constructJavaBean2ndStep(MappingNode node, Object object) {
223             Class<? extends Object> beanType = node.getType();
224             List<NodeTuple> nodeValue = (List<NodeTuple>) node.getValue();
225             for (NodeTuple tuple : nodeValue) {
226                 ScalarNode keyNode;
227                 if (tuple.getKeyNode() instanceof ScalarNode) {
228                     // key must be scalar
229                     keyNode = (ScalarNode) tuple.getKeyNode();
230                 } else {
231                     throw new YAMLException("Keys must be scalars but found: " + tuple.getKeyNode());
232                 }
233                 Node valueNode = tuple.getValueNode();
234                 // keys can only be Strings
235                 keyNode.setType(String.class);
236                 String key = (String) constructObject(keyNode);
237                 boolean isArray = false;
238                 try {
239                     Property property = getProperty(beanType, key);
240                     valueNode.setType(property.getType());
241                     if (property.getType().isArray()) {
242                         isArray = true;
243                     }
244                     TypeDescription memberDescription = typeDefinitions.get(beanType);
245                     boolean typeDetected = false;
246                     if (memberDescription != null) {
247                         switch (valueNode.getNodeId()) {
248                         case sequence:
249                             SequenceNode snode = (SequenceNode) valueNode;
250                             Class<? extends Object> memberType = memberDescription
251                                     .getListPropertyType(key);
252                             if (memberType != null) {
253                                 snode.setListType(memberType);
254                                 typeDetected = true;
255                             } else if (property.getType().isArray()) {
256                                 isArray = true;
257                                 snode.setListType(property.getType().getComponentType());
258                                 typeDetected = true;
259                             }
260                             break;
261                         case mapping:
262                             MappingNode mnode = (MappingNode) valueNode;
263                             Class<? extends Object> keyType = memberDescription.getMapKeyType(key);
264                             if (keyType != null) {
265                                 mnode.setKeyType(keyType);
266                                 mnode.setValueType(memberDescription.getMapValueType(key));
267                                 typeDetected = true;
268                             }
269                             break;
270                         }
271                     }
272                     if (!typeDetected && valueNode.getNodeId() != NodeId.scalar) {
273                         // only if there is no explicit TypeDescription
274                         Type[] arguments = property.getActualTypeArguments();
275                         if (arguments != null) {
276                             // type safe (generic) collection may contain the
277                             // proper class
278                             if (valueNode.getNodeId() == NodeId.sequence) {
279                                 Class t = (Class) arguments[0];
280                                 SequenceNode snode = (SequenceNode) valueNode;
281                                 snode.setListType(t);
282                             } else if (Tags.SET.equals(valueNode.getTag())) {
283                                 Class t = (Class) arguments[0];
284                                 MappingNode mnode = (MappingNode) valueNode;
285                                 mnode.setKeyType(t);
286                                 mnode.setUseClassConstructor(true);
287                             } else if (valueNode.getNodeId() == NodeId.mapping) {
288                                 Class ketType = (Class) arguments[0];
289                                 Class valueType = (Class) arguments[1];
290                                 MappingNode mnode = (MappingNode) valueNode;
291                                 mnode.setKeyType(ketType);
292                                 mnode.setValueType(valueType);
293                                 mnode.setUseClassConstructor(true);
294                             }
295                         }
296                     }
297                     Object value = constructObject(valueNode);
298                     if (isArray && value instanceof List) {
299                         List<Object> list = (List<Object>) value;
300                         value = list.toArray(createArray(property.getType()));
301                     }
302                     property.set(object, value);
303                 } catch (Exception e) {
304                     throw new YAMLException("Cannot create property=" + key + " for JavaBean="
305                             + object + "; " + e.getMessage(), e);
306                 }
307             }
308             return object;
309         }

312         private <T> T[] createArray(Class<T> type) {
313             return (T[]) Array.newInstance(type.getComponentType(), 0);
314         }

316         private Property getProperty(Class<? extends Object> type, String name)
317                 throws IntrospectionException {
318             // check cache
319             Map<String, Property> properties = propertiesCache.get(type);
320             if (properties == null) {
321                 properties = new HashMap<String, Property>();
322                 propertiesCache.put(type, properties);
323                 for (PropertyDescriptor property : Introspector.getBeanInfo(type)
324                         .getPropertyDescriptors()) {
325                     String methodName = property.getName();
326                     if (property.getWriteMethod() != null) {
327                         properties.put(methodName, new MethodProperty(property));
328                     }
329                 }
330                 for (Field field : type.getFields()) {
331                     int modifiers = field.getModifiers();
332                     if (Modifier.isStatic(modifiers) || Modifier.isTransient(modifiers)) {
333                         continue;
334                     }
335                     properties.put(field.getName(), new FieldProperty(field));
336                 }
337             }
338             // take from cache
339             Property property = properties.get(name);
340             if (property == null) {
341                 throw new YAMLException("Unable to find property '" + name + "' on class: "
342                         + type.getName());
343             } else {
344                 return property;
345             }
346         }

357         @SuppressWarnings("unchecked")
358         private Construct getConstructor(Node node) {
359             Class cl = getClassForNode(node);
360             node.setType(cl);
361             // call the constructor as if the runtime class is defined
362             Construct constructor = yamlClassConstructors.get(node.getNodeId());
363             return constructor;

366         public Object construct(Node node) {
367             Object result = null;
368             try {
369                 result = getConstructor(node).construct(node);
370             } catch (Exception e) {
371                 throw new ConstructorException(null, null, "Can't construct a java object for "
372                         + node.getTag() + "; exception=" + e.getMessage(), node.getStartMark(), e);
373             }
374             return result;

388     /***
389      * Construct scalar instance when the runtime class is known. Recursive
390      * structures are not supported.
391      */
392     protected class ConstructScalar extends AbstractConstruct {
393         @SuppressWarnings("unchecked")
394         public Object construct(Node nnode) {
395             ScalarNode node = (ScalarNode) nnode;
396             Class type = node.getType();
397             Object result;
398             if (type.isPrimitive() || type == String.class || Number.class.isAssignableFrom(type)
399                     || type == Boolean.class || Date.class.isAssignableFrom(type)
400                     || type == Character.class || type == BigInteger.class
401                     || Enum.class.isAssignableFrom(type) || Tags.BINARY.equals(node.getTag())) {
402                 // standard classes created directly
403                 result = constructStandardJavaInstance(type, node);
404             } else {
405                 // there must be only 1 constructor with 1 argument
406                 java.lang.reflect.Constructor[] javaConstructors = type.getConstructors();
407                 int oneArgCount = 0;
408                 java.lang.reflect.Constructor javaConstructor = null;
409                 for (java.lang.reflect.Constructor c : javaConstructors) {
410                     if (c.getParameterTypes().length == 1) {
411                         oneArgCount++;
412                         javaConstructor = c;
413                     }
414                 }
415                 Object argument;
416                 if (javaConstructor == null) {
417                     throw new YAMLException("No single argument constructor found for " + type);
418                 } else if (oneArgCount == 1) {
419                     argument = constructStandardJavaInstance(
420                             javaConstructor.getParameterTypes()[0], node);
421                 } else {
422                     // TODO it should be possible to use implicit types instead
423                     // of forcing String. Resolver must be available here to
424                     // obtain the implicit tag. Then we can set the tag and call
425                     // callConstructor(node) to create the argument instance.
426                     // On the other hand it may be safer to require a custom
427                     // constructor to avoid guessing the argument class
428                     argument = constructScalar(node);
429                     try {
430                         javaConstructor = type.getConstructor(String.class);
431                     } catch (Exception e) {
432                         throw new ConstructorException(null, null,
433                                 "Can't construct a java object for scalar " + node.getTag()
434                                         + "; No String constructor found. Exception="
435                                         + e.getMessage(), node.getStartMark(), e);
436                     }
437                 }
438                 try {
439                     result = javaConstructor.newInstance(argument);
440                 } catch (Exception e) {
441                     throw new ConstructorException(null, null,
442                             "Can't construct a java object for scalar " + node.getTag()
443                                     + "; exception=" + e.getMessage(), node.getStartMark(), e);
444                 }
445             }
446             return result;
447         }

449         @SuppressWarnings("unchecked")
450         private Object constructStandardJavaInstance(Class type, ScalarNode node) {
451             Object result;
452             if (type == String.class) {
453                 Construct stringConstructor = yamlConstructors.get(Tags.STR);
454                 result = stringConstructor.construct((ScalarNode) node);
455             } else if (type == Boolean.class || type == Boolean.TYPE) {
456                 Construct boolConstructor = yamlConstructors.get(Tags.BOOL);
457                 result = boolConstructor.construct((ScalarNode) node);
458             } else if (type == Character.class || type == Character.TYPE) {
459                 Construct charConstructor = yamlConstructors.get(Tags.STR);
460                 String ch = (String) charConstructor.construct((ScalarNode) node);
461                 if (ch.length() == 0) {
462                     result = null;
463                 } else if (ch.length() != 1) {
464                     throw new YAMLException("Invalid node Character: '" + ch + "'; length: "
465                             + ch.length());
466                 } else {
467                     result = new Character(ch.charAt(0));
468                 }
469             } else if (Date.class.isAssignableFrom(type)) {
470                 Construct dateConstructor = yamlConstructors.get(Tags.TIMESTAMP);
471                 Date date = (Date) dateConstructor.construct((ScalarNode) node);
472                 if (type == Date.class) {
473                     result = date;
474                 } else {
475                     try {
476                         java.lang.reflect.Constructor<?> constr = type.getConstructor(long.class);
477                         result = constr.newInstance(date.getTime());
478                     } catch (Exception e) {
479                         throw new YAMLException("Cannot construct: '" + type + "'");
480                     }
481                 }
482             } else if (type == Float.class || type == Double.class || type == Float.TYPE
483                     || type == Double.TYPE || type == BigDecimal.class) {
484                 Construct doubleConstructor = yamlConstructors.get(Tags.FLOAT);
485                 result = doubleConstructor.construct(node);
486                 if (type == Float.class || type == Float.TYPE) {
487                     result = new Float((Double) result);
488                 } else if (type == BigDecimal.class) {
489                     result = new BigDecimal(((Double) result).doubleValue());
490                 }
491             } else if (type == Byte.class || type == Short.class || type == Integer.class
492                     || type == Long.class || type == BigInteger.class || type == Byte.TYPE
493                     || type == Short.TYPE || type == Integer.TYPE || type == Long.TYPE) {
494                 Construct intConstructor = yamlConstructors.get(Tags.INT);
495                 result = intConstructor.construct(node);
496                 if (type == Byte.class || type == Byte.TYPE) {
497                     result = new Byte(result.toString());
498                 } else if (type == Short.class || type == Short.TYPE) {
499                     result = new Short(result.toString());
500                 } else if (type == Integer.class || type == Integer.TYPE) {
501                     result = new Integer(result.toString());
502                 } else if (type == Long.class || type == Long.TYPE) {
503                     result = new Long(result.toString());
504                 } else {
505                     // only BigInteger left
506                     result = new BigInteger(result.toString());
507                 }
508             } else if (Enum.class.isAssignableFrom(type)) {
509                 String enumValueName = node.getValue();
510                 try {
511                     result = Enum.valueOf(type, enumValueName);
512                 } catch (Exception ex) {
513                     throw new YAMLException("Unable to find enum value '" + enumValueName
514                             + "' for enum class: " + type.getName());
515                 }
516             } else {
517                 throw new YAMLException("Unsupported class: " + type);
518             }
519             return result;
520         }

523     /***
524      * Construct sequence (List, Array, or immutable object) when the runtime
525      * class is known.
526      */
527     private class ConstructSequence implements Construct {
528         @SuppressWarnings("unchecked")
529         public Object construct(Node node) {
530             SequenceNode snode = (SequenceNode) node;
531             if (List.class.isAssignableFrom(node.getType()) || node.getType().isArray()) {
532                 if (node.isTwoStepsConstruction()) {
533                     return createDefaultList(snode.getValue().size());
534                 } else {
535                     return constructSequence(snode);
536                 }
537             } else {
538                 // create immutable object
539                 List<java.lang.reflect.Constructor> possibleConstructors = new ArrayList<java.lang.reflect.Constructor>(
540                         snode.getValue().size());
541                 for (java.lang.reflect.Constructor constructor : node.getType().getConstructors()) {
542                     if (snode.getValue().size() == constructor.getParameterTypes().length) {
543                         possibleConstructors.add(constructor);
544                     }
545                 }
546                 if (possibleConstructors.isEmpty()) {
547                     throw new YAMLException("No constructors with "
548                             + String.valueOf(snode.getValue().size()) + " arguments found for "
549                             + node.getType());
550                 }
551                 List<Object> argumentList;
552                 if (possibleConstructors.size() == 1) {
553                     argumentList = new ArrayList<Object>(snode.getValue().size());
554                     java.lang.reflect.Constructor c = possibleConstructors.get(0);
555                     int index = 0;
556                     for (Node argumentNode : snode.getValue()) {
557                         Class type = c.getParameterTypes()[index];
558                         // set runtime classes for arguments
559                         argumentNode.setType(type);
560                         Object argumentValue = constructObject(argumentNode);
561                         argumentList.add(argumentValue);
562                         index++;
563                     }
564                 } else {
565                     // use BaseConstructor
566                     argumentList = (List<Object>) constructSequence(snode);
567                 }
568                 Class[] parameterTypes = new Class[argumentList.size()];
569                 int index = 0;
570                 for (Object parameter : argumentList) {
571                     parameterTypes[index] = parameter.getClass();
572                     index++;
573                 }
574                 java.lang.reflect.Constructor javaConstructor;
575                 try {
576                     Class cl = node.getType();
577                     javaConstructor = cl.getConstructor(parameterTypes);
578                     Object[] initargs = argumentList.toArray();
579                     return javaConstructor.newInstance(initargs);
580                 } catch (Exception e) {
581                     throw new YAMLException(e);
582                 }
583             }
584         }

587         public void construct2ndStep(Node node, Object object) {
588             SequenceNode snode = (SequenceNode) node;
589             List<Object> list = (List<Object>) object;

598     protected Class<?> getClassForNode(Node node) {
599         Class<? extends Object> classForTag = typeTags.get(node.getTag());
600         if (classForTag == null) {
601             if (node.getTag().length() < Tags.PREFIX.length()) {
602                 throw new YAMLException("Unknown tag: " + node.getTag());
603             }
604             String name = node.getTag().substring(Tags.PREFIX.length());
605             Class<?> cl;
606             try {
607                 cl = getClassForName(name);
608             } catch (ClassNotFoundException e) {
609                 throw new YAMLException("Class not found: " + name);
610             }
611             typeTags.put(node.getTag(), cl);
612             return cl;
613         } else {
614             return classForTag;
615         }
616     }
