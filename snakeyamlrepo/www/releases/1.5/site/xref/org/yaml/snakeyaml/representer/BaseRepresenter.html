37  /***
38   * Represent basic YAML structures: scalar, sequence, mapping
39   * 
40   * @see <a href="http://pyyaml.org/wiki/PyYAML">PyYAML</a> for more information
41   */
42  public abstract class BaseRepresenter {
43      @SuppressWarnings("unchecked")
44      protected final Map<Class, Represent> representers = new HashMap<Class, Represent>();
45      /***
46       * in Java 'null' is not a type. So we have to keep the null representer
47       * separately otherwise it will coincide with the default representer which
48       * is stored with the key null.
49       */
50      protected Represent nullRepresenter;
51      @SuppressWarnings("unchecked")
52      protected final Map<Class, Represent> multiRepresenters = new HashMap<Class, Represent>();
53      private Character defaultStyle;
54      protected Boolean defaultFlowStyle;
55      protected final Map<Object, Node> representedObjects = new IdentityHashMap<Object, Node>();
56      private final Set<Object> objectKeeper = new HashSet<Object>();
57      protected Object objectToRepresent;
58  
59      public void represent(Serializer serializer, Object data) throws IOException {
60          Node node = representData(data);
61          serializer.serialize(node);
62          representedObjects.clear();
63          objectKeeper.clear();
64          objectToRepresent = null;
65      }
66  
67      @SuppressWarnings("unchecked")
68      protected Node representData(Object data) {
69          objectToRepresent = data;
70          if (!ignoreAliases(data)) {
71              // check for identity
72              if (representedObjects.containsKey(objectToRepresent)) {
73                  Node node = representedObjects.get(objectToRepresent);
74                  return node;
75              }
76          }
77          // check for null first
78          if (data == null) {
79              Node node = nullRepresenter.representData(data);
80              return node;
81          }
82          // check the same class
83          Node node;
84          Class clazz = data.getClass();
85          if (representers.containsKey(clazz)) {
86              Represent representer = representers.get(clazz);
87              node = representer.representData(data);
88          } else {
89              // check the parents
90              for (Class repr : multiRepresenters.keySet()) {
91                  if (repr.isInstance(data)) {
92                      Represent representer = multiRepresenters.get(repr);
93                      node = representer.representData(data);
94                      return node;
95                  }
96              }
97              // check array of primitives
98              if (clazz.isArray()) {
99                  throw new YAMLException("Arrays of primitives are not fully supported.");
100             }
101             // check defaults
102             if (multiRepresenters.containsKey(null)) {
103                 Represent representer = multiRepresenters.get(null);
104                 node = representer.representData(data);
105             } else {
106                 Represent representer = representers.get(null);
107                 node = representer.representData(data);
108             }
109         }
110         return node;
111     }
112 
113     protected Node representScalar(String tag, String value, Character style) {
114         if (style == null) {
115             style = this.defaultStyle;
116         }
117         Node node = new ScalarNode(tag, value, null, null, style);
118         representedObjects.put(objectToRepresent, node);
119         return node;
120     }
121 
122     protected Node representScalar(String tag, String value) {
123         return representScalar(tag, value, null);
124     }
125 
126     protected Node representSequence(String tag, List<? extends Object> sequence, Boolean flowStyle) {
127         List<Node> value = new ArrayList<Node>(sequence.size());
128         SequenceNode node = new SequenceNode(tag, value, flowStyle);
129         representedObjects.put(objectToRepresent, node);
130         boolean bestStyle = true;
131         for (Object item : sequence) {
132             Node nodeItem = representData(item);
133             if (!((nodeItem instanceof ScalarNode && ((ScalarNode) nodeItem).getStyle() == null))) {
134                 bestStyle = false;
135             }
136             value.add(nodeItem);
137         }
138         if (flowStyle == null) {
139             if (defaultFlowStyle != null) {
140                 node.setFlowStyle(defaultFlowStyle);
141             } else {
142                 node.setFlowStyle(bestStyle);
143             }
144         }
145         return node;
146     }
147 
148     protected Node representMapping(String tag, Map<? extends Object, Object> mapping,
149             Boolean flowStyle) {
150         List<NodeTuple> value = new ArrayList<NodeTuple>(mapping.size());
151         MappingNode node = new MappingNode(tag, value, flowStyle);
152         representedObjects.put(objectToRepresent, node);
153         boolean bestStyle = true;
154         for (Object itemKey : mapping.keySet()) {
155             Object itemValue = mapping.get(itemKey);
156             Node nodeKey = representData(itemKey);
157             Node nodeValue = representData(itemValue);
158             if (!((nodeKey instanceof ScalarNode && ((ScalarNode) nodeKey).getStyle() == null))) {
159                 bestStyle = false;
160             }
161             if (!((nodeValue instanceof ScalarNode && ((ScalarNode) nodeValue).getStyle() == null))) {
162                 bestStyle = false;
163             }
164             value.add(new NodeTuple(nodeKey, nodeValue));
165         }
166         if (flowStyle == null) {
167             if (defaultFlowStyle != null) {
168                 node.setFlowStyle(defaultFlowStyle);
169             } else {
170                 node.setFlowStyle(bestStyle);
171             }
172         }
173         return node;
174     }
175 
176     protected abstract boolean ignoreAliases(Object data);
177 
178     public void setDefaultScalarStyle(ScalarStyle defaultStyle) {
179         this.defaultStyle = defaultStyle.getChar();
180     }
181 
182     public void setDefaultFlowStyle(FlowStyle defaultFlowStyle) {
183         this.defaultFlowStyle = defaultFlowStyle.getStyleBoolean();
184     }
