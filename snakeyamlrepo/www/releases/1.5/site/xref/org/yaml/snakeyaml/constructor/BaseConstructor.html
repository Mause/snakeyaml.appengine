42  public abstract class BaseConstructor {
43      /***
44       * It maps the node kind to the the Construct implementation. When the
45       * runtime class is known then the implicit tag is ignored.
46       */
47      protected final Map<NodeId, Construct> yamlClassConstructors = new EnumMap<NodeId, Construct>(
48              NodeId.class);
49      /***
50       * It maps the (explicit or implicit) tag to the Construct implementation.
51       * It is used: <br/>
52       * 1) explicit tag - if present. <br/>
53       * 2) implicit tag - when the runtime class of the instance is unknown (the
54       * node has the Object.class)
55       */
56      protected final Map<String, Construct> yamlConstructors = new HashMap<String, Construct>();
57  
58      private Composer composer;
59      private final Map<Node, Object> constructedObjects;
60      private final Set<Node> recursiveObjects;
61      private final ArrayStack<Tuple<Node, Object>> toBeConstructedAt2ndStep;
62      private final ArrayList<Tuple<Map<Object, Object>, Tuple<Object, Object>>> maps2fill;
63      private final ArrayList<Tuple<Set<Object>, Object>> sets2fill;

65      protected String rootTag;

67      public BaseConstructor() {
68          constructedObjects = new HashMap<Node, Object>();
69          recursiveObjects = new HashSet<Node>();
70          toBeConstructedAt2ndStep = new ArrayStack<Tuple<Node, Object>>(10);
71          maps2fill = new ArrayList<Tuple<Map<Object, Object>, Tuple<Object, Object>>>();
72          sets2fill = new ArrayList<Tuple<Set<Object>, Object>>();
73          rootTag = null;
74      }

81       * Check if more documents available

90      /***
91       * Construct and return the next document
92       * 
93       * @return constructed instance
94       */
95      public Object getData() {
96          // Construct and return the next document.
97          composer.checkNode();
98          Node node = composer.getNode();
99          if (rootTag != null) {
100             node.setTag(rootTag);
101         }
102         return constructDocument(node);
103     }

105     /***
106      * Ensure that the stream contains a single document and construct it
107      * 
108      * @return constructed instance
109      * @throws ComposerException
110      *             in case there are more documents in the stream
111      */
112     public Object getSingleData() {
113         // Ensure that the stream contains a single document and construct it
114         Node node = composer.getSingleNode();
115         if (node != null) {
116             if (rootTag != null) {
117                 node.setTag(rootTag);
118             }
119             return constructDocument(node);
120         }
121         return null;
122     }

124     /***
125      * Construct complete YAML document. Call the second step in case of
126      * recursive structures. At the end cleans all the state.
127      * 
128      * @param node
129      *            root Node
130      * @return Java instance
131      */
132     private Object constructDocument(Node node) {
133         Object data = constructObject(node);
134         while (!toBeConstructedAt2ndStep.isEmpty()) {
135             Tuple<Node, Object> toBeProcessed = toBeConstructedAt2ndStep.pop();
136             callPostCreate(toBeProcessed._1(), toBeProcessed._2());
137         }
138         if (!maps2fill.isEmpty()) {
139             for (Tuple<Map<Object, Object>, Tuple<Object, Object>> entry : maps2fill) {
140                 Tuple<Object, Object> key_value = entry._2();
141                 entry._1().put(key_value._1(), key_value._2());
142             }
143             maps2fill.clear();
144         }
145         if (!sets2fill.isEmpty()) {
146             for (Tuple<Set<Object>, Object> value : sets2fill) {
147                 value._1().add(value._2());
148             }
149             sets2fill.clear();
150         }
151         constructedObjects.clear();
152         recursiveObjects.clear();
153         toBeConstructedAt2ndStep.clear();
154         return data;
155     }

157     /***
158      * Construct object from the specified Node. Return existing instance if the
159      * node is already constructed.
160      * 
161      * @param node
162      *            Node to be constructed
163      * @return Java instance
164      */
165     protected Object constructObject(Node node) {
166         if (constructedObjects.containsKey(node)) {
167             return constructedObjects.get(node);
168         }
169         if (recursiveObjects.contains(node)) {
170             throw new ConstructorException(null, null, "found unconstructable recursive node", node
171                     .getStartMark());
172         }
173         recursiveObjects.add(node);
174         Object data = callConstructor(node);
175         if (node.isTwoStepsConstruction()) {
176             toBeConstructedAt2ndStep.push(new Tuple<Node, Object>(node, data));
177         }
178         constructedObjects.put(node, data);
179         recursiveObjects.remove(node);
180         return data;
181     }

183     /***
184      * Create Java instance from the specified Node
185      * 
186      * @param node
187      *            Node to be constructed
188      * @return Java instance
189      */
190     protected Object callConstructor(Node node) {
191         return getConstructor(node).construct(node);
192     }
193 
194     protected void callPostCreate(Node node, Object object) {
195         getConstructor(node).construct2ndStep(node, object);
196     }

198     /***
199      * Get the constructor to construct the Node. For implicit tags if the
200      * runtime class is known a dedicated Construct implementation is used.
201      * Otherwise the constructor is chosen by the tag.
202      * 
203      * @param node
204      *            Node to be constructed
205      * @return Construct implementation for the specified node
206      */
207     private Construct getConstructor(Node node) {
208         if (node.useClassConstructor()) {
209             return yamlClassConstructors.get(node.getNodeId());
210         } else {
211             Construct constructor = yamlConstructors.get(node.getTag());
212             if (constructor == null) {
213                 return yamlConstructors.get(null);
214             }
215             return constructor;
216         }
217     }

227     @SuppressWarnings("unchecked")
228     protected List<? extends Object> constructSequence(SequenceNode node) {
229         List<Object> result;
230         if (List.class.isAssignableFrom(node.getType()) && !node.getType().isInterface()) {
231             // the root class may be defined (Vector for instance)
232             try {
233                 result = (List<Object>) node.getType().newInstance();
234             } catch (Exception e) {
235                 throw new YAMLException(e);
236             }
237         } else {
238             result = createDefaultList(node.getValue().size());
239         }
240         constructSequenceStep2(node, result);
241         return result;
242     }
243 
244     protected void constructSequenceStep2(SequenceNode node, List<Object> list) {
245         for (Node child : node.getValue()) {
246             list.add(constructObject(child));
247         }
248     }

272     protected void constructMapping2ndStep(MappingNode node, Map<Object, Object> mapping) {
273         List<NodeTuple> nodeValue = (List<NodeTuple>) node.getValue();
274         for (NodeTuple tuple : nodeValue) {
275             Node keyNode = tuple.getKeyNode();
276             Node valueNode = tuple.getValueNode();
277             Object key = constructObject(keyNode);
278             if (key != null) {
279                 try {
280                     key.hashCode();// check circular dependencies
281                 } catch (Exception e) {
282                     throw new ConstructorException("while constructing a mapping", node
283                             .getStartMark(), "found unacceptable key " + key, tuple.getKeyNode()
284                             .getStartMark(), e);
285                 }
286             }
287             Object value = constructObject(valueNode);
288             if (keyNode.isTwoStepsConstruction()) {
289                 /*
290                  * if keyObject is created it 2 steps we should postpone putting
291                  * it in map because it may have different hash after
292                  * initialization compared to clean just created one. And map of
293                  * course does not observe key hashCode changes.
294                  */
295                 maps2fill.add(0, new Tuple<Map<Object, Object>, Tuple<Object, Object>>(mapping,
296                         new Tuple<Object, Object>(key, value)));
297             } else {
298                 mapping.put(key, value);
299             }
300         }
301     }
302 
303     protected void constructSet2ndStep(MappingNode node, Set<Object> set) {
304         List<NodeTuple> nodeValue = (List<NodeTuple>) node.getValue();
305         for (NodeTuple tuple : nodeValue) {
306             Node keyNode = tuple.getKeyNode();
307             Object key = constructObject(keyNode);
308             if (key != null) {
309                 try {
310                     key.hashCode();// check circular dependencies
311                 } catch (Exception e) {
312                     throw new ConstructorException("while constructing a Set", node.getStartMark(),
313                             "found unacceptable key " + key, tuple.getKeyNode().getStartMark(), e);
314                 }
315             }
316             if (keyNode.isTwoStepsConstruction()) {
317                 /*
318                  * if keyObject is created it 2 steps we should postpone putting
319                  * it into the set because it may have different hash after
320                  * initialization compared to clean just created one. And set of
321                  * course does not observe value hashCode changes.
322                  */
323                 sets2fill.add(0, new Tuple<Set<Object>, Object>(set, key));
324             } else {
325                 set.add(key);
326             }
327         }
328     }
