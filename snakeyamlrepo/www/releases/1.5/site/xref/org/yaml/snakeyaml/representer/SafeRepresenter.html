34  /***
35   * Represent standard Java classes
36   * 
37   * @see <a href="http://pyyaml.org/wiki/PyYAML">PyYAML</a> for more information

39  class SafeRepresenter extends BaseRepresenter {
40  
41      protected Map<Class<? extends Object>, String> classTags;
42  
43      public SafeRepresenter() {
44          this.nullRepresenter = new RepresentNull();
45          this.representers.put(String.class, new RepresentString());
46          this.representers.put(Boolean.class, new RepresentBoolean());
47          this.representers.put(Character.class, new RepresentString());
48          this.representers.put(byte[].class, new RepresentByteArray());
49          this.multiRepresenters.put(Number.class, new RepresentNumber());
50          this.multiRepresenters.put(List.class, new RepresentList());
51          this.multiRepresenters.put(Map.class, new RepresentMap());
52          this.multiRepresenters.put(Set.class, new RepresentSet());
53          this.multiRepresenters.put(new Object[0].getClass(), new RepresentArray());
54          this.multiRepresenters.put(Date.class, new RepresentDate());
55          this.multiRepresenters.put(Enum.class, new RepresentEnum());
56          classTags = new HashMap<Class<? extends Object>, String>();
57      }
58  
59      private String getTag(Class<?> clazz, String defaultTag) {
60          if (classTags.containsKey(clazz)) {
61              return classTags.get(clazz);
62          } else {
63              return defaultTag;
64          }
65      }
66  
67      @Override
68      protected boolean ignoreAliases(Object data) {
69          if (data == null) {
70              return true;
71          }
72          if (data instanceof Object[]) {
73              Object[] array = (Object[]) data;
74              if (array.length == 0) {
75                  return true;
76              }
77          }
78          return data instanceof String || data instanceof Boolean || data instanceof Integer
79                  || data instanceof Long || data instanceof Float || data instanceof Double
80                  || data instanceof Enum;
81      }
82  
83      /***
84       * Define a tag for the <code>Class</code> to serialize
85       * 
86       * @param clazz
87       *            <code>Class</code> which tag is changed
88       * @param tag
89       *            new tag to be used for every instance of the specified
90       *            <code>Class</code>
91       * @return the previous tag associated with the <code>Class</code>
92       */
93      public String addClassTag(Class<? extends Object> clazz, String tag) {
94          if (tag == null) {
95              throw new NullPointerException("Tag must be provided.");
96          }
97          return classTags.put(clazz, tag);
98      }

106     public static Pattern BINARY_PATTERN = Pattern.compile("[//x00-//x08//x0B//x0C//x0E-//x1F]");
107 
108     private class RepresentString implements Represent {
109         public Node representData(Object data) {
110             String tag = Tags.STR;
111             Character style = null;
112             String value = data.toString();
113             if (BINARY_PATTERN.matcher(value).find()) {
114                 tag = Tags.BINARY;
115                 char[] binary;
116                 binary = Base64Coder.encode(value.getBytes());
117                 value = String.valueOf(binary);
118                 style = '|';
119             }
120             return representScalar(tag, value, style);
121         }
122     }

124     private class RepresentBoolean implements Represent {
125         public Node representData(Object data) {
126             String value;
127             if (Boolean.TRUE.equals(data)) {
128                 value = "true";
129             } else {
130                 value = "false";
131             }
132             return representScalar(Tags.BOOL, value);
133         }
134     }

136     private class RepresentNumber implements Represent {
137         public Node representData(Object data) {
138             String tag;
139             String value;
140             if (data instanceof Byte || data instanceof Short || data instanceof Integer
141                     || data instanceof Long || data instanceof BigInteger) {
142                 tag = Tags.INT;
143                 value = data.toString();
144             } else {
145                 Number number = (Number) data;
146                 tag = Tags.FLOAT;
147                 if (number.equals(Double.NaN)) {
148                     value = ".NaN";
149                 } else if (number.equals(Double.POSITIVE_INFINITY)) {
150                     value = ".inf";
151                 } else if (number.equals(Double.NEGATIVE_INFINITY)) {
152                     value = "-.inf";
153                 } else {
154                     value = number.toString();
155                 }
156             }
157             return representScalar(getTag(data.getClass(), tag), value);
158         }
159     }

184     private class RepresentSet implements Represent {
185         @SuppressWarnings("unchecked")
186         public Node representData(Object data) {
187             Map<Object, Object> value = new LinkedHashMap<Object, Object>();
188             Set<Object> set = (Set<Object>) data;
189             for (Object key : set) {

192             return representMapping(getTag(data.getClass(), Tags.SET), value, null);

196     private class RepresentDate implements Represent {
197         public Node representData(Object data) {
198             // because SimpleDateFormat ignores timezone we have to use Calendar
199             Calendar calendar = Calendar.getInstance(TimeZone.getTimeZone("UTC"));
200             calendar.setTime((Date) data);
201             int years = calendar.get(Calendar.YEAR);
202             int months = calendar.get(Calendar.MONTH) + 1; // 0..12
203             int days = calendar.get(Calendar.DAY_OF_MONTH); // 1..31
204             int hour24 = calendar.get(Calendar.HOUR_OF_DAY); // 0..24
205             int minutes = calendar.get(Calendar.MINUTE); // 0..59
206             int seconds = calendar.get(Calendar.SECOND); // 0..59
207             int millis = calendar.get(Calendar.MILLISECOND);
208             StringBuffer buffer = new StringBuffer(String.valueOf(years));
209             buffer.append("-");
210             if (months < 10) {
211                 buffer.append("0");
212             }
213             buffer.append(String.valueOf(months));
214             buffer.append("-");
215             if (days < 10) {
216                 buffer.append("0");
217             }
218             buffer.append(String.valueOf(days));
219             buffer.append("T");
220             if (hour24 < 10) {
221                 buffer.append("0");
222             }
223             buffer.append(String.valueOf(hour24));
224             buffer.append(":");
225             if (minutes < 10) {
226                 buffer.append("0");
227             }
228             buffer.append(String.valueOf(minutes));
229             buffer.append(":");
230             if (seconds < 10) {
231                 buffer.append("0");
232             }
233             buffer.append(String.valueOf(seconds));
234             if (millis > 0) {
235                 if (millis < 10) {
236                     buffer.append(".00");
237                 } else if (millis < 100) {
238                     buffer.append(".0");
239                 } else {
240                     buffer.append(".");
241                 }
242                 buffer.append(String.valueOf(millis));
243             }
244             buffer.append("Z");
245             return representScalar(getTag(data.getClass(), Tags.TIMESTAMP), buffer.toString(), null);
246         }
247     }
