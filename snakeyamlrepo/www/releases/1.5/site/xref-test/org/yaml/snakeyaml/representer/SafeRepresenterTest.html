83      public void testStyle() {
84          List<Integer> list = new ArrayList<Integer>();
85          list.add(new Integer(1));
86          list.add(new Integer(1));
87          Map<String, Object> map = new HashMap<String, Object>();
88          map.put("list", list);
89          map.put("name", "Ubuntu");
90          map.put("age", 5);
91          DumperOptions options = new DumperOptions();
92          options.setDefaultScalarStyle(DumperOptions.ScalarStyle.DOUBLE_QUOTED);
93          options.setDefaultFlowStyle(DumperOptions.FlowStyle.BLOCK);
94          Yaml yaml = new Yaml(options);
95          String output = yaml.dump(map);
96          assertTrue(output.contains("\"age\": !!int \"5\""));
97          assertTrue(output.contains("\"name\": \"Ubuntu\""));
98          assertTrue(output.contains("- !!int \"1\""));
99      }

101     public void testStyle2() {

112         Yaml yaml = new Yaml(options);
