18  import java.util.LinkedHashMap;

27  public class EmitterTest extends TestCase {
28  
29      public void testWriteFolded() {
30          DumperOptions options = new DumperOptions();
31          options.setDefaultScalarStyle(ScalarStyle.FOLDED);
32          String folded = "0123456789 0123456789\n0123456789 0123456789";
33          Map<String, String> map = new LinkedHashMap<String, String>();
34          map.put("aaa", folded);
35          map.put("bbb", "\nbla-bla\n");
36          Yaml yaml = new Yaml(options);
37          String output = yaml.dump(map);
38          String etalon = "\"aaa\": >-\n  0123456789 0123456789\n\n  0123456789 0123456789\n\"bbb\": >2\n\n  bla-bla\n";
39          assertEquals(etalon, output);
40      }
