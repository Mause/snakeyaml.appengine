54      public void testNoRootTag() throws IOException {
55          CarWithWheel car1 = new CarWithWheel();
56          car1.setPlate("12-XP-F4");
57          Wheel wheel = new Wheel();
58          wheel.setId(2);
59          car1.setWheel(wheel);
60          Map<String, Integer> map = new HashMap<String, Integer>();
61          map.put("id", 3);
62          car1.setMap(map);
63          car1.setYear("2008");
64          DumperOptions options = new DumperOptions();
65          options.setExplicitRoot(Tags.MAP);
66          String carYaml1 = new Yaml(options).dump(car1);
67          assertEquals(Util.getLocalResource("constructor/car-without-root-tag.yaml"), carYaml1);
68          //
69          Constructor contructor = new Constructor(CarWithWheel.class);
70          Loader loader = new Loader(contructor);
71          CarWithWheel car2 = (CarWithWheel) new Yaml(loader).load(carYaml1);
72          String carYaml2 = new Yaml(options).dump(car2);
73          assertEquals(carYaml1, carYaml2);
74      }

93      public void testLoadClassTag() throws IOException {
94          Constructor constructor = new Constructor();
95          constructor.addTypeDescription(new TypeDescription(Car.class, "!car"));
96          Loader loader = new Loader(constructor);
97          Yaml yaml = new Yaml(loader);
98          Car car = (Car) yaml.load(Util.getLocalResource("constructor/car-without-tags.yaml"));
99          assertEquals("12-XP-F4", car.getPlate());
100         List<Wheel> wheels = car.getWheels();
101         assertNotNull(wheels);
102         assertEquals(5, wheels.size());
103         Wheel w1 = wheels.get(0);
104         assertEquals(1, w1.getId());
105         //
106         String carYaml1 = new Yaml().dump(car);
107         assertTrue(carYaml1.startsWith("!!org.yaml.snakeyaml.constructor.Car"));
108         //
109         Representer representer = new Representer();
110         representer.addClassTag(Car.class, "!car");
111         Dumper dumper = new Dumper(representer, new DumperOptions());
112         yaml = new Yaml(dumper);
113         String carYaml2 = yaml.dump(car);
114         assertEquals(Util.getLocalResource("constructor/car-without-tags.yaml"), carYaml2);
115     }
