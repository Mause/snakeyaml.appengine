32  public class Constructor extends SafeConstructor {
33      private Map<String, Class<? extends Object>> typeTags;
34      private Map<Class<? extends Object>, TypeDescription> typeDefinitions;
35  
36      public Constructor() {
37          this(Object.class);
38      }
39  
40      public Constructor(Class<? extends Object> theRoot) {
41          if (theRoot == null) {
42              throw new NullPointerException("Root type must be provided.");
43          }
44          this.yamlConstructors.put(null, new ConstuctYamlObject());
45          rootType = theRoot;
46          typeTags = new HashMap<String, Class<? extends Object>>();
47          typeDefinitions = new HashMap<Class<? extends Object>, TypeDescription>();
48      }

50      /***
51       * Make YAML aware how to parse a custom Class. If there is no root Class
52       * assigned in constructor then the 'root' property of this definition is
53       * respected.
54       * 
55       * @param definition
56       *            to be added to the Constructor
57       * @return the previous value associated with <tt>definition</tt>, or
58       *         <tt>null</tt> if there was no mapping for <tt>definition</tt>.
59       */
60      public TypeDescription addTypeDescription(TypeDescription definition) {
61          if (definition == null) {
62              throw new NullPointerException("TypeDescription is required.");
63          }
64          if (rootType == Object.class && definition.isRoot()) {
65              rootType = definition.getType();
66          }
67          String tag = definition.getTag();
68          typeTags.put(tag, definition.getType());
69          return typeDefinitions.put(definition.getType(), definition);
70      }

72      private class ConstuctYamlObject implements Construct {
73          @SuppressWarnings("unchecked")
74          public Object construct(Node node) {
75              Object result = null;
76              Class<? extends Object> customTag = typeTags.get(node.getTag());
77              try {
78                  Class cl;
79                  if (customTag == null) {
80                      if (node.getTag().length() < "tag:yaml.org,2002:".length()) {
81                          throw new YAMLException("Unknown tag: " + node.getTag());
82                      }
83                      String name = node.getTag().substring("tag:yaml.org,2002:".length());
84                      cl = Class.forName(name);
85                  } else {
86                      cl = customTag;
87                  }
88                  if (node instanceof MappingNode) {
89                      MappingNode mnode = (MappingNode) node;
90                      mnode.setType(cl);
91                      result = constructMappingNode(mnode);
92                  } else if (node instanceof SequenceNode) {
93                      SequenceNode snode = (SequenceNode) node;
94                      List<Object> values = (List<Object>) constructSequence(snode);
95                      Class[] parameterTypes = new Class[values.size()];
96                      int index = 0;
97                      for (Object parameter : values) {
98                          parameterTypes[index] = parameter.getClass();
99                          index++;
100                     }
101                     java.lang.reflect.Constructor javaConstructor = cl
102                             .getConstructor(parameterTypes);
103                     Object[] initargs = values.toArray();
104                     result = javaConstructor.newInstance(initargs);
105                 } else {
106                     ScalarNode snode = (ScalarNode) node;
107                     Object value = constructScalar(snode);
108                     java.lang.reflect.Constructor javaConstructor = cl.getConstructor(value
109                             .getClass());
110                     result = javaConstructor.newInstance(value);
111                 }
112             } catch (Exception e) {
113                 throw new ConstructorException(null, null, "Can't construct a java object for "
114                         + node.getTag() + "; exception=" + e.getMessage(), node.getStartMark());
115             }
116             return result;
117         }
118     }

120     @Override
121     protected Object callConstructor(Node node) {
122         if (Object.class.equals(node.getType())) {
123             return super.callConstructor(node);
124         }
125         Object result;
126         if (node instanceof ScalarNode) {
127             result = constructScalarNode((ScalarNode) node);
128         } else if (node instanceof SequenceNode) {
129             result = constructSequence((SequenceNode) node);
130         } else {
131             if (Map.class.isAssignableFrom(node.getType())) {
132                 result = super.constructMapping((MappingNode) node);
133             } else {
134                 result = constructMappingNode((MappingNode) node);
135             }
136         }
137         return result;
138     }

140     private Object constructScalarNode(ScalarNode node) {
141         Class<? extends Object> type = node.getType();
142         Object result;
143         if (type.isPrimitive() || type == String.class || Number.class.isAssignableFrom(type)
144                 || type == Boolean.class || type == Date.class || type == Character.class
145                 || type == BigInteger.class) {
146             if (type == String.class) {
147                 Construct stringContructor = yamlConstructors.get("tag:yaml.org,2002:str");
148                 result = stringContructor.construct((ScalarNode) node);
149             } else if (type == Boolean.class || type == Boolean.TYPE) {
150                 Construct boolContructor = yamlConstructors.get("tag:yaml.org,2002:bool");
151                 result = boolContructor.construct((ScalarNode) node);
152             } else if (type == Character.class || type == Character.TYPE) {
153                 Construct charContructor = yamlConstructors.get("tag:yaml.org,2002:str");
154                 String ch = (String) charContructor.construct((ScalarNode) node);
155                 if (ch.length() != 1) {
156                     throw new YAMLException("Invalid node Character: '" + ch + "'; length: "
157                             + ch.length());
158                 }
159                 result = new Character(ch.charAt(0));
160             } else if (type == Date.class) {
161                 Construct dateContructor = yamlConstructors.get("tag:yaml.org,2002:timestamp");
162                 result = dateContructor.construct((ScalarNode) node);
163             } else if (type == Float.class || type == Double.class || type == Float.TYPE
164                     || type == Double.TYPE) {
165                 Construct doubleContructor = yamlConstructors.get("tag:yaml.org,2002:float");
166                 result = doubleContructor.construct(node);
167                 if (type == Float.class || type == Float.TYPE) {
168                     result = new Float((Double) result);
169                 }
170             } else if (Number.class.isAssignableFrom(type) || type == Byte.TYPE
171                     || type == Short.TYPE || type == Integer.TYPE || type == Long.TYPE) {
172                 Construct intContructor = yamlConstructors.get("tag:yaml.org,2002:int");
173                 result = intContructor.construct(node);
174                 if (type == Byte.class || type == Byte.TYPE) {
175                     result = new Byte(result.toString());
176                 } else if (type == Short.class || type == Short.TYPE) {
177                     result = new Short(result.toString());
178                 } else if (type == Integer.class || type == Integer.TYPE) {
179                     result = new Integer(result.toString());
180                 } else if (type == Long.class || type == Long.TYPE) {
181                     result = new Long(result.toString());
182                 } else if (type == BigInteger.class) {
183                     result = new BigInteger(result.toString());
184                 } else {
185                     throw new YAMLException("Unsupported Number class: " + type);
186                 }
187             } else {
188                 throw new YAMLException("Unsupported class: " + type);
189             }
190         } else {
191             try {
192                 // get value by BaseConstructor
193                 Object value = super.callConstructor(node);
194                 java.lang.reflect.Constructor<? extends Object> javaConstructor = type
195                         .getConstructor(value.getClass());
196                 result = javaConstructor.newInstance(value);
197             } catch (Exception e) {
198                 throw new YAMLException(e);
199             }
200         }
201         return result;
202     }

204     /***
205      * Construct JavaBean. If type safe collections are used please look at
206      * <code>TypeDescription</code>.
207      * 
208      * @param node
209      *            - node where the keys are property names (they can only be
210      *            <code>String</code>s) and values are objects to be created
211      * @return constructed JavaBean
212      */
213     @SuppressWarnings("unchecked")
214     private Object constructMappingNode(MappingNode node) {
215         Class<? extends Object> beanType = node.getType();
216         Object object;
217         try {
218             object = beanType.newInstance();
219         } catch (InstantiationException e) {
220             throw new YAMLException(e);
221         } catch (IllegalAccessException e) {
222             throw new YAMLException(e);
223         }
224         List<Node[]> nodeValue = (List<Node[]>) node.getValue();
225         for (Iterator<Node[]> iter = nodeValue.iterator(); iter.hasNext();) {
226             Node[] tuple = iter.next();
227             ScalarNode keyNode;
228             if (tuple[0] instanceof ScalarNode) {
229                 keyNode = (ScalarNode) tuple[0];// key must be scalar
230             } else {
231                 throw new YAMLException("Keys must be scalars but found: " + tuple[0]);
232             }
233             Node valueNode = tuple[1];
234             // keys can only be Strings
235             keyNode.setType(String.class);
236             String key = (String) constructObject(keyNode);
237             boolean isArray = false;
238             try {
239                 Property property = getProperty(beanType, key);
240                 if (property == null)
241                     throw new YAMLException("Unable to find property '" + key + "' on class: "
242                             + beanType.getName());
243                 valueNode.setType(property.getType());
244                 TypeDescription memberDescription = typeDefinitions.get(beanType);
245                 if (memberDescription != null) {
246                     if (valueNode instanceof SequenceNode) {
247                         SequenceNode snode = (SequenceNode) valueNode;
248                         Class<? extends Object> memberType = memberDescription
249                                 .getListPropertyType(key);
250                         if (memberType != null) {
251                             snode.setListType(memberType);
252                         } else if (property.getType().isArray()) {
253                             isArray = true;
254                             snode.setListType(property.getType().getComponentType());
255                         }
256                     } else if (valueNode instanceof MappingNode) {
257                         MappingNode mnode = (MappingNode) valueNode;
258                         Class<? extends Object> keyType = memberDescription.getMapKeyType(key);
259                         if (keyType != null) {
260                             mnode.setKeyType(keyType);
261                             mnode.setValueType(memberDescription.getMapValueType(key));
262                         }
263                     }
264                 }
265                 Object value = constructObject(valueNode);
266                 if (isArray) {
267                     List<Object> list = (List<Object>) value;
268                     value = list.toArray(createArray(property.getType()));
269                 }
270                 property.set(object, value);
271             } catch (Exception e) {
272                 throw new YAMLException(e);
273             }
274         }
275         return object;
276     }

278     @SuppressWarnings("unchecked")
279     private <T> T[] createArray(Class<T> type) {
280         return (T[]) Array.newInstance(type.getComponentType(), 0);
281     }

283     protected Property getProperty(Class<? extends Object> type, String name)
284             throws IntrospectionException {
285         for (PropertyDescriptor property : Introspector.getBeanInfo(type).getPropertyDescriptors()) {
286             if (property.getName().equals(name)) {
287                 if (property.getReadMethod() != null && property.getWriteMethod() != null)
288                     return new MethodProperty(property);
289                 break;
290             }
291         }
292         for (Field field : type.getFields()) {
293             int modifiers = field.getModifiers();
294             if (!Modifier.isPublic(modifiers) || Modifier.isStatic(modifiers)
295                     || Modifier.isTransient(modifiers))
296                 continue;
297             if (field.getName().equals(name))
298                 return new FieldProperty(field);
299         }
300         return null;
301     }
