8   import org.yaml.snakeyaml.composer.Composer;
9   import org.yaml.snakeyaml.constructor.BaseConstructor;
10  import org.yaml.snakeyaml.constructor.Constructor;
11  import org.yaml.snakeyaml.parser.ParserImpl;
12  import org.yaml.snakeyaml.reader.Reader;

15  /***
16   * @see <a href="http://pyyaml.org/wiki/PyYAML">PyYAML</a> for more information
17   */
18  public class Loader {
19      protected BaseConstructor constructor;
20      protected Resolver resolver;
21  
22      public Loader(BaseConstructor constructor) {
23          super();
24          this.constructor = constructor;
25          this.resolver = new Resolver();
26      }
27  
28      public Loader() {
29          this(new Constructor());
30      }
31  
32      public Object load(java.io.Reader io) {
33          Composer composer = new Composer(new ParserImpl(new Reader(io)), resolver);
34          constructor.setComposer(composer);
35          return constructor.getSingleData();
36      }
37  
38      public Iterable<Object> loadAll(java.io.Reader yaml) {
39          Composer composer = new Composer(new ParserImpl(new Reader(yaml)), resolver);
40          this.constructor.setComposer(composer);
41          Iterator<Object> result = new Iterator<Object>() {
42              public boolean hasNext() {
43                  return constructor.checkData();
44              }
45  
46              public Object next() {
47                  return constructor.getData();
48              }
49  
50              public void remove() {
51                  throw new UnsupportedOperationException();
52              }
53          };
54          return new YamlIterable(result);
55      }
56  
57      private class YamlIterable implements Iterable<Object> {
58          private Iterator<Object> iterator;
59  
60          public YamlIterable(Iterator<Object> iterator) {
61              this.iterator = iterator;
62          }
63  
64          public Iterator<Object> iterator() {
65              return iterator;
66          }
67  
68      }
69  
70      public void setResolver(Resolver resolver) {
71          this.resolver = resolver;
72      }
73  }
