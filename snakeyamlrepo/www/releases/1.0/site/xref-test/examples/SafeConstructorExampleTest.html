15  public class SafeConstructorExampleTest extends TestCase {
16      @SuppressWarnings("unchecked")
17      public void testConstruct() throws IOException {
18          String doc = "- 5\n- Person\n- true";
19          Loader loader = new Loader(new SafeConstructor());
20          Yaml yaml = new Yaml(loader);
21          List<Object> list = (List<Object>) yaml.load(doc);
22          System.out.println(list);
23          assertEquals(3, list.size());
24          assertEquals(new Integer(5), list.get(0));
25          assertEquals("Person", list.get(1));
26          assertEquals(Boolean.TRUE, list.get(2));
27      }

29      public void testSafeConstruct() throws IOException {
30          String doc = "- 5\n- !org.yaml.snakeyaml.constructor.Person\n  firstName: Andrey\n  age: 99\n- true";
31          Loader loader = new Loader(new SafeConstructor());
32          Yaml yaml = new Yaml(loader);
33          try {
34              yaml.load(doc);
35              fail("Custom Java classes should not be created.");
36          } catch (Exception e) {
37              assertEquals(
38                      "null; could not determine a constructor for the tag !org.yaml.snakeyaml.constructor.Person",
39                      e.getMessage());
40          }
41      }
