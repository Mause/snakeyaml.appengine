40      private Object construct(Constructor constructor, String data) {
41          Reader reader = new Reader(data);
42          Parser parser = new ParserImpl(reader);
43          Resolver resolver = new Resolver();
44          Composer composer = new Composer(parser, resolver);
45          constructor.setComposer(composer);
46          return constructor.getSingleData();
47      }
