13  public class BoolTagTest extends AbstractTest {
14      public void testBool() throws IOException {
15          assertEquals(Boolean.TRUE, getMapValue("canonical: true", "canonical"));
16          assertEquals(Boolean.FALSE, getMapValue("answer: NO", "answer"));
17          assertEquals(Boolean.TRUE, getMapValue("logical: True", "logical"));
18          assertEquals(Boolean.TRUE, getMapValue("option: on", "option"));
19      }
20  
21      public void testBoolCanonical() throws IOException {
22          assertEquals(Boolean.TRUE, getMapValue("canonical: Yes", "canonical"));
23          assertEquals(Boolean.TRUE, getMapValue("canonical: yes", "canonical"));
24          assertEquals(Boolean.TRUE, getMapValue("canonical: YES", "canonical"));
25          assertEquals("yES", getMapValue("canonical: yES", "canonical"));
26          assertEquals(Boolean.FALSE, getMapValue("canonical: No", "canonical"));
27          assertEquals(Boolean.FALSE, getMapValue("canonical: NO", "canonical"));
28          assertEquals(Boolean.FALSE, getMapValue("canonical: no", "canonical"));
29          assertEquals(Boolean.FALSE, getMapValue("canonical: off", "canonical"));
30          assertEquals(Boolean.FALSE, getMapValue("canonical: Off", "canonical"));
31          assertEquals(Boolean.FALSE, getMapValue("canonical: OFF", "canonical"));
32          assertEquals(Boolean.TRUE, getMapValue("canonical: ON", "canonical"));
33          assertEquals(Boolean.TRUE, getMapValue("canonical: On", "canonical"));
34          assertEquals(Boolean.TRUE, getMapValue("canonical: on", "canonical"));
35          // it looks like it is against the specification but it is like in
36          // PyYAML
37          assertEquals("n", getMapValue("canonical: n", "canonical"));
38          assertEquals("N", getMapValue("canonical: N", "canonical"));
39          assertEquals("y", getMapValue("canonical: y", "canonical"));
40          assertEquals("Y", getMapValue("canonical: Y", "canonical"));
41      }
