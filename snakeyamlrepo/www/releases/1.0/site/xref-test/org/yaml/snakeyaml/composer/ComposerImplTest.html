15  public class ComposerImplTest extends TestCase {
16  
17      public void testGetNode() {
18          String data = "american:\n  - Boston Red Sox";
19          Node node = compose(data);
20          assertNotNull(node);
21          assertTrue(node instanceof MappingNode);
22          // System.out.println(node);
23          String data2 = "---\namerican:\n- Boston Red Sox";
24          Node node2 = compose(data2);
25          assertNotNull(node2);
26          assertFalse(node.equals(node2));
27      }
28  
29      private Node compose(String data) {
30          Reader reader = new Reader(data);
31          Parser parser = new ParserImpl(reader);
32          Resolver resolver = new Resolver();
33          Composer composer = new Composer(parser, resolver);
34          Node node = composer.getSingleNode();
35          return node;
36      }
