24      @SuppressWarnings("unchecked")
25      public void testGetToken() {
26          String data = "string: abcd";
27          Reader reader = new Reader(data);
28          Scanner scanner = new ScannerImpl(reader);
29  
30          Mark dummy = new Mark("dummy", 0, 0, 0, "", 0);
31          LinkedList<Token> etalonTokens = new LinkedList<Token>();
32          etalonTokens.add(new StreamStartToken(dummy, dummy));
33          etalonTokens.add(new BlockMappingStartToken(dummy, dummy));
34          etalonTokens.add(new KeyToken(dummy, dummy));
35          etalonTokens.add(new ScalarToken("string", true, dummy, dummy, (char) 0));
36          etalonTokens.add(new ValueToken(dummy, dummy));
37          etalonTokens.add(new ScalarToken("abcd", true, dummy, dummy, (char) 0));
38          etalonTokens.add(new BlockEndToken(dummy, dummy));
39          etalonTokens.add(new StreamEndToken(dummy, dummy));
40          while (scanner.checkToken(new ArrayList())) {
41              assertEquals(etalonTokens.removeFirst(), scanner.getToken());
42              // System.out.println(scanner.getToken());
43          }
44          assertFalse("Must contain no more tokens: " + scanner.getToken(), scanner
45                  .checkToken(new ArrayList()));
46      }
