25      @SuppressWarnings("unchecked")
26      public void testAddImplicitResolver() {
27          Dumper dumper = new Dumper(new MyRepresenter(), new DumperOptions());
28          Loader loader = new Loader(new MyConstructor());
29          Yaml yaml = new Yaml(loader, dumper);
30          Pattern regexp = Pattern.compile("//d//d-//d//d-//d//d//d");
31          yaml.addImplicitResolver("tag:yaml.org,2002:Phone", regexp, "0123456789");
32          Phone phone1 = new Phone("12-34-567");
33          Phone phone2 = new Phone("11-22-333");
34          Phone phone3 = new Phone("44-55-777");
35          List<Phone> etalonList = new LinkedList<Phone>();
36          etalonList.add(phone1);
37          etalonList.add(phone2);
38          etalonList.add(phone3);
39          String output = yaml.dump(etalonList);
40          assertEquals("[12-34-567, 11-22-333, 44-55-777]\n", output);
41          List<Phone> parsedList = (List<Phone>) yaml.load(output);
42          assertEquals(3, parsedList.size());
43          assertEquals(phone1, parsedList.get(0));
44          assertEquals(phone2, parsedList.get(1));
45          assertEquals(phone3, parsedList.get(2));
46          assertEquals(etalonList, parsedList);
47      }
