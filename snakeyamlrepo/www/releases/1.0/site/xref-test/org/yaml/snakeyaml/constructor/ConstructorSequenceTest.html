53      private List<Object> construct(String data) {

57      @SuppressWarnings("unchecked")
58      private List<Object> construct(Constructor constructor, String data) {
59          Reader reader = new Reader(data);
60          Parser parser = new ParserImpl(reader);
61          Resolver resolver = new Resolver();
62          Composer composer = new Composer(parser, resolver);
63          constructor.setComposer(composer);
64          List result = (List) constructor.getSingleData();
65          return result;
66      }
