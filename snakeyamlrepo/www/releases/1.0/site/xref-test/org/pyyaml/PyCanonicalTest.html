30      private List<Token> canonicalScan(InputStream input) throws IOException {
31          int ch = input.read();
32          StringBuffer buffer = new StringBuffer();
33          while (ch != -1) {
34              buffer.append((char) ch);
35              ch = input.read();
36          }
37          CanonicalScanner scanner = new CanonicalScanner(buffer.toString());
38          List<Token> result = new LinkedList<Token>();
39          while (scanner.peekToken() != null) {
40              result.add(scanner.getToken());
41          }
42          return result;
43      }
