25  public class PyErrorsTest extends PyImportTest {
26      private boolean skip(String filename) {
27          List<String> failures = new ArrayList<String>();
28          // in python list cannot be a key in a dictionary.
29          failures.add("unacceptable-key.loader-error");
30          for (String name : failures) {
31              if (name.equals(filename)) {
32                  return true;
33              }
34          }
35          return false;
36      }

38      public void testLoaderErrors() throws FileNotFoundException {
39          File[] files = getStreamsByExtension(".loader-error");
40          assertTrue("No test files found.", files.length > 0);
41          for (int i = 0; i < files.length; i++) {
42              if (skip(files[i].getName())) {
43                  continue;
44              }
45              try {
46                  for (Object document : loadAll(new FileInputStream(files[i]))) {
47                      assertNotNull("File " + files[i], document);
48                  }
49                  fail("Loading must fail for " + files[i].getAbsolutePath());
50                  // System.err.println("Loading must fail for " +
51                  // files[i].getAbsolutePath());
52              } catch (Exception e) {
53                  assertTrue(true);
54              }
55          }
56      }
