9    * @see imported from PyYAML

11  public class PyMarkTest extends PyImportTest {
12  
13      public void testMarks() {
14          String content = getResource("test_mark.marks");
15          String[] inputs = content.split("---\n");
16          for (int i = 1; i < inputs.length; i++) {
17              String input = inputs[i];
18              int index = 0;
19              int line = 0;
20              int column = 0;
21              while (input.charAt(index) != '*') {
22                  if (input.charAt(index) != '\n') {
23                      line += 1;
24                      column = 0;
25                  } else {
26                      column += 1;
27                  }
28                  index += 1;
29              }
30              Mark mark = new Mark("testMarks", index, line, column, input, index);
31              String snippet = mark.get_snippet(2, 79);
32              assertTrue("Must only have one '\n'.", snippet.indexOf("\n") > -1);
33              assertEquals("Must only have only one '\n'.", snippet.indexOf("\n"), snippet
34                      .lastIndexOf("\n"));
35              String[] lines = snippet.split("\n");
36              String data = lines[0];
37              String pointer = lines[1];
38              assertTrue("Mark must be restricted: " + data, data.length() < 82);
39              int dataPosition = data.indexOf("*");
40              int pointerPosition = pointer.indexOf("^");
41              assertEquals("Pointer should coincide with '*':\n " + snippet, dataPosition,
42                      pointerPosition);
43          }
44      }
