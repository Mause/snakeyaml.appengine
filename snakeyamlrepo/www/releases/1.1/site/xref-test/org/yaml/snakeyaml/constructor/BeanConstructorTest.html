19      public void testPrimitivesConstructor() throws IOException {
20          Loader loader = new Loader(new Constructor(TestBean1.class));
21          Yaml yaml = new Yaml(loader);
22          String document = Util.getLocalResource("constructor/test-primitives1.yaml");
23          System.out.println(document);
24          TestBean1 result = (TestBean1) yaml.load(document);
25          assertNotNull(result);
26          assertEquals(new Byte((byte) 1), result.getByteClass());
27          assertEquals((byte) -3, result.getBytePrimitive());
28          assertEquals(new Short((short) 0), result.getShortClass());
29          assertEquals((short) -13, result.getShortPrimitive());
30          assertEquals(new Integer(5), result.getInteger());
31          assertEquals(17, result.getIntPrimitive());
32          assertEquals("the text", result.getText());
33          assertEquals("13", result.getId());
34          assertEquals(new Long(11111111111L), result.getLongClass());
35          assertEquals(9999999999L, result.getLongPrimitive());
36          assertEquals(Boolean.TRUE, result.getBooleanClass());
37          assertTrue(result.isBooleanPrimitive());
38          assertEquals(new Character('2'), result.getCharClass());
39          assertEquals('#', result.getCharPrimitive());
40          assertEquals(new BigInteger("1234567890123456789012345678901234567890"), result
41                  .getBigInteger());
42          assertEquals(new Float(2), result.getFloatClass());
43          assertEquals(new Float(3.1416), result.getFloatPrimitive());
44          assertEquals(new Double(4), result.getDoubleClass());
45          assertEquals(new Double(11200), result.getDoublePrimitive());
46          assertEquals(1199836800000L, result.getDate().getTime());
47          assertEquals("public", result.publicField);
48          //
49          DumperOptions options = new DumperOptions();
50          options.setDefaultFlowStyle(DumperOptions.DefaultFlowStyle.BLOCK);
51          Yaml yamlToDump = new Yaml(options);
52          String output = yamlToDump.dump(result);
53          System.out.println(output);
54          TestBean1 result2 = (TestBean1) yaml.load(output);
55          assertNotNull(result2);
56          TestBean1 result3 = (TestBean1) new Yaml().load(output);
57          assertNotNull(result3);
58      }
