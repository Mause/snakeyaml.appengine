29      public EventConstructor() {
30          this.yamlConstructors.put(null, new ConstructEvent());

33      private class ConstructEvent implements Construct {
34  
35          @SuppressWarnings("unchecked")
36          public Object construct(Node node) {
37              Map mapping;
38              if (node instanceof ScalarNode) {
39                  mapping = new HashMap();
40              } else {
41                  mapping = constructMapping((MappingNode) node);
42              }
43              String className = node.getTag().substring(1) + "Event";
44              Event value;
45              if (className.equals("AliasEvent")) {
46                  value = new AliasEvent((String) mapping.get("anchor"), null, null);
47              } else if (className.equals("ScalarEvent")) {
48                  String tag = (String) mapping.get("tag");
49                  String v = (String) mapping.get("value");
50                  if (v == null) {
51                      v = "";
52                  }
53                  List implicitList = (List) mapping.get("implicit");
54                  boolean implicit[];
55                  if (implicitList == null) {
56                      implicit = new boolean[] { false, true };
57                  } else {
58                      implicit = new boolean[2];
59                      implicit[0] = (Boolean) implicitList.get(0);
60                      implicit[1] = (Boolean) implicitList.get(1);
61                  }
62                  value = new ScalarEvent((String) mapping.get("anchor"), tag, implicit, v, null,
63                          null, null);
64              } else if (className.equals("SequenceStartEvent")) {
65                  String tag = (String) mapping.get("tag");
66                  Boolean implicit = (Boolean) mapping.get("implicit");
67                  if (implicit == null) {
68                      implicit = true;
69                  }
70                  value = new SequenceStartEvent((String) mapping.get("anchor"), tag, implicit, null,
71                          null, false);
72              } else if (className.equals("MappingStartEvent")) {
73                  String tag = (String) mapping.get("tag");
74                  Boolean implicit = (Boolean) mapping.get("implicit");
75                  if (implicit == null) {
76                      implicit = true;
77                  }
78                  value = new MappingStartEvent((String) mapping.get("anchor"), tag, implicit, null,
79                          null, false);
80              } else if (className.equals("DocumentEndEvent")) {
81                  value = new DocumentEndEvent(null, null, false);
82              } else if (className.equals("DocumentStartEvent")) {
83                  Map<String, String> tags = (Map<String, String>) mapping.get("tags");
84                  List<Integer> versionList = (List<Integer>) mapping.get("version");
85                  Integer[] version = null;
86                  if (versionList != null) {
87                      version = new Integer[2];
88                      version[0] = versionList.get(0).intValue();
89                      version[1] = versionList.get(1).intValue();
90                  }
91                  value = new DocumentStartEvent(null, null, false, version, tags);
92              } else if (className.equals("MappingEndEvent")) {
93                  value = new MappingEndEvent(null, null);
94              } else if (className.equals("SequenceEndEvent")) {
95                  value = new SequenceEndEvent(null, null);
96              } else if (className.equals("StreamEndEvent")) {
97                  value = new StreamEndEvent(null, null);
98              } else if (className.equals("StreamStartEvent")) {
99                  value = new StreamStartEvent(null, null);
100             } else {
101                 throw new UnsupportedOperationException();
102             }
103             return value;
104         }
