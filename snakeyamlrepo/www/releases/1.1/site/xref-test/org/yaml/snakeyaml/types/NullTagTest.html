13  public class NullTagTest extends AbstractTest {

15      public void testNull() throws IOException {
16          assertNull("Got: '" + load("---\n") + "'", load("---\n"));
17          assertNull(load("---\n..."));
18          assertNull(load("---\n...\n"));
19          assertNull(load("\n"));
20          assertNull(load(""));
21          assertNull(load(" "));
22          assertNull(load("~"));
23          assertNull(load("---\n~"));
24          assertNull(load("null"));
25          assertNull(load("Null"));
26          assertNull(load("NULL"));
27          assertNull(getMapValue("empty:\n", "empty"));
28          assertNull(getMapValue("canonical: ~", "canonical"));
29          assertNull(getMapValue("english: null", "english"));
30          assertNull(getMapValue("english: Null", "english"));
31          assertNull(getMapValue("english: NULL", "english"));
32          assertEquals("null key", getMapValue("~: null key\n", null));
33      }

35      @SuppressWarnings("unchecked")
36      public void testSequenceNull() throws IOException {
37          String input = "---\n# This sequence has five\n# entries, two have values.\nsparse:\n  - ~\n  - 2nd entry\n  -\n  - 4th entry\n  - Null\n";
38          List<String> parsed = (List<String>) getMapValue(input, "sparse");
39          assertEquals(5, parsed.size());
40          assertNull(parsed.get(0));
41          assertEquals("2nd entry", parsed.get(1));
42          assertNull("Got: '" + parsed.get(2) + "'", parsed.get(2));
43          assertEquals("4th entry", parsed.get(3));
44          assertNull(parsed.get(4));
45      }

47      public void testNullInMap() throws IOException {
48          String input = "key1: null\n~: value1";
49          Map<String, Object> parsed = getMap(input);
