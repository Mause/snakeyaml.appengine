46      private void _testEmitter(String mask, boolean canonical) throws IOException {
47          File[] files = getStreamsByExtension(mask, true);
48          assertTrue("No test files found.", files.length > 0);
49          for (File file : files) {
50              // if (!file.getName().contains("spec-06-01.canonical")) {
51              // continue;
52              // }
53              try {
54                  List<Event> events = parse(new FileInputStream(file));
55                  //
56                  StringWriter stream = new StringWriter();
57                  DumperOptions options = new DumperOptions();
58                  options.setCanonical(canonical);
59                  Emitter emitter = new Emitter(stream, options);
60                  for (Event event : events) {
61                      emitter.emit(event);
62                  }
63                  //
64                  String data = stream.toString();
65                  List<Event> newEvents = new LinkedList<Event>();
66                  Reader reader = new Reader(data);
67                  Parser parser = new ParserImpl(reader);
68                  while (parser.peekEvent() != null) {
69                      Event event = parser.getEvent();
70                      newEvents.add(event);
71                  }
72                  // check
73                  assertEquals(events.size(), newEvents.size());
74                  Iterator<Event> iter1 = events.iterator();
75                  Iterator<Event> iter2 = newEvents.iterator();
76                  while (iter1.hasNext()) {
77                      Event event = iter1.next();
78                      Event newEvent = iter2.next();
79                      assertEquals(event.getClass().getName(), newEvent.getClass().getName());
80                      if (event instanceof NodeEvent) {
81                          NodeEvent e1 = (NodeEvent) event;
82                          NodeEvent e2 = (NodeEvent) newEvent;
83                          assertEquals(e1.getAnchor(), e2.getAnchor());
84                      }
85                      if (event instanceof CollectionStartEvent) {
86                          CollectionStartEvent e1 = (CollectionStartEvent) event;
87                          CollectionStartEvent e2 = (CollectionStartEvent) newEvent;
88                          assertEquals(e1.getTag(), e2.getTag());
89                      }
90                      if (event instanceof ScalarEvent) {
91                          ScalarEvent e1 = (ScalarEvent) event;
92                          ScalarEvent e2 = (ScalarEvent) newEvent;
93                          boolean[] implicit1 = e1.getImplicit();
94                          boolean[] implicit2 = e2.getImplicit();
95                          if (!implicit1[0] && !implicit1[1] && !implicit2[0] && !implicit2[1]) {
96                              assertEquals(e1.getTag(), e2.getTag());
97                          }
98                          assertEquals(e1.getValue(), e2.getValue());
99                      }
100                 }
101             } catch (Exception e) {
102                 System.out.println("Failed File: " + file);
103                 // fail("Failed File: " + file + "; " + e.getMessage());
104                 throw new RuntimeException(e);
105             }
106         }

111         File[] canonicalFiles = getStreamsByExtension(".canonical", false);
112         assertTrue("No test files found.", canonicalFiles.length > 0);
113         File[] dataFiles = getStreamsByExtension(".data", true);
114         assertTrue("No test files found.", dataFiles.length > 0);
115         List<File> allFiles = new LinkedList(Arrays.asList(canonicalFiles));
116         allFiles.addAll(Arrays.asList(dataFiles));
117         for (File file : allFiles) {
118             try {
119                 List<Event> events = new LinkedList<Event>();
120                 Reader reader = new Reader(new UnicodeReader(new FileInputStream(file)));
121                 Parser parser = new ParserImpl(reader);
122                 while (parser.peekEvent() != null) {
123                     Event event = parser.getEvent();
124                     events.add(event);
125                 }
126                 //
127                 for (Boolean flowStyle : new Boolean[] { Boolean.FALSE, Boolean.TRUE }) {
128                     for (DumperOptions.DefaultScalarStyle style : DumperOptions.DefaultScalarStyle
129                             .values()) {
130                         List<Event> styledEvents = new LinkedList<Event>();
131                         for (Event event : events) {
132                             if (event instanceof ScalarEvent) {
133                                 ScalarEvent scalar = (ScalarEvent) event;
134                                 event = new ScalarEvent(scalar.getAnchor(), scalar.getTag(), scalar
135                                         .getImplicit(), scalar.getValue(), scalar.getStartMark(),
136                                         scalar.getEndMark(), style.getChar());
137                             } else if (event instanceof SequenceStartEvent) {
138                                 SequenceStartEvent seqStart = (SequenceStartEvent) event;
139                                 event = new SequenceStartEvent(seqStart.getAnchor(), seqStart
140                                         .getTag(), seqStart.getImplicit(), seqStart.getStartMark(),
141                                         seqStart.getEndMark(), flowStyle);
142                             } else if (event instanceof MappingStartEvent) {
143                                 MappingStartEvent mapStart = (MappingStartEvent) event;
144                                 event = new MappingStartEvent(mapStart.getAnchor(), mapStart
145                                         .getTag(), mapStart.getImplicit(), mapStart.getStartMark(),
146                                         mapStart.getEndMark(), flowStyle);
147                             }
148                             styledEvents.add(event);
149                         }
150                         // emit
151                         String data = emit(styledEvents);
152                         List<Event> newEvents = parse(data);
153                         assertEquals("Events must not change. File: " + file, events.size(),
154                                 newEvents.size());
155                         Iterator<Event> oldIter = events.iterator();
156                         Iterator<Event> newIter = newEvents.iterator();
157                         while (oldIter.hasNext()) {
158                             Event event = oldIter.next();
159                             Event newEvent = newIter.next();
160                             assertEquals(event.getClass(), newEvent.getClass());
161                             if (event instanceof NodeEvent) {
162                                 assertEquals(((NodeEvent) event).getAnchor(),
163                                         ((NodeEvent) newEvent).getAnchor());
164                             }
165                             if (event instanceof CollectionStartEvent) {
166                                 assertEquals(((CollectionStartEvent) event).getTag(),
167                                         ((CollectionStartEvent) newEvent).getTag());
168                             }
169                             if (event instanceof ScalarEvent) {
170                                 ScalarEvent scalarOld = (ScalarEvent) event;
171                                 ScalarEvent scalarNew = (ScalarEvent) newEvent;
172                                 boolean[] oldImplicit = scalarOld.getImplicit();
173                                 boolean[] newImplicit = scalarNew.getImplicit();
174                                 if (!oldImplicit[0] && !oldImplicit[1] && !newImplicit[0]
175                                         && !newImplicit[1]) {
176                                     assertEquals(scalarOld.getTag(), scalarNew.getTag());
177                                 }
178                                 assertEquals(scalarOld.getValue(), scalarNew.getValue());
179                             }
180                         }
181                     }

192     private String emit(List<Event> events) throws IOException {
193         StringWriter writer = new StringWriter();
194         Emitter emitter = new Emitter(writer, new DumperOptions());
195         for (Event event : events) {
196             emitter.emit(event);
197         }
198         return writer.toString();
199     }

201     private List<Event> parse(String data) {
202         ParserImpl parser = new ParserImpl(new Reader(data));
203         List<Event> newEvents = new LinkedList<Event>();
204         while (parser.peekEvent() != null) {
205             newEvents.add(parser.getEvent());
206         }
207         return newEvents;
