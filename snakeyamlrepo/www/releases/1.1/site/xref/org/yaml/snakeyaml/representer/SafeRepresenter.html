25      public SafeRepresenter(Character default_style, Boolean default_flow_style) {
26          super(default_style, default_flow_style);
27          this.nullRepresenter = new RepresentNull();
28          this.representers.put(String.class, new RepresentString());
29          this.representers.put(Boolean.class, new RepresentBoolean());
30          this.representers.put(Character.class, new RepresentString());
31          this.representers.put(byte[].class, new RepresentByteArray());
32          this.multiRepresenters.put(Number.class, new RepresentNumber());
33          this.multiRepresenters.put(List.class, new RepresentList());
34          this.multiRepresenters.put(Map.class, new RepresentMap());
35          this.multiRepresenters.put(Set.class, new RepresentSet());
36          this.multiRepresenters.put(new Object[0].getClass(), new RepresentArray());
37          this.multiRepresenters.put(Date.class, new RepresentDate());
38          this.multiRepresenters.put(Enum.class, new RepresentEnum());
39      }

41      @Override
42      protected boolean ignoreAliases(Object data) {
43          if (data == null) {
44              return true;
45          }
46          if (data instanceof Object[]) {
47              Object[] array = (Object[]) data;
48              if (array.length == 0) {
49                  return true;
50              }
51          }
52          return data instanceof String || data instanceof Boolean || data instanceof Integer
53                  || data instanceof Long || data instanceof Float || data instanceof Double
54                  || data instanceof Enum;
55      }

63      public static Pattern BINARY_PATTERN = Pattern.compile("[//x00-//x08//x0B//x0C//x0E-//x1F]");
64  
65      private class RepresentString implements Represent {
66          public Node representData(Object data) {
67              String tag = "tag:yaml.org,2002:str";
68              Character style = null;
69              String value = data.toString();
70              if (BINARY_PATTERN.matcher(value).find()) {
71                  tag = "tag:yaml.org,2002:binary";
72                  char[] binary;
73                  binary = Base64Coder.encode(value.getBytes());
74                  value = String.valueOf(binary);
75                  style = '|';
76              }
77              return representScalar(tag, value, style);
78          }
79      }

81      private class RepresentBoolean implements Represent {
82          public Node representData(Object data) {
83              String value;
84              if (Boolean.TRUE.equals(data)) {
85                  value = "true";
86              } else {
87                  value = "false";
88              }
89              return representScalar("tag:yaml.org,2002:bool", value);
90          }
91      }

93      private class RepresentNumber implements Represent {
94          public Node representData(Object data) {
95              String tag;
96              String value;
97              if (data instanceof Byte || data instanceof Short || data instanceof Integer
98                      || data instanceof Long || data instanceof BigInteger) {
99                  tag = "tag:yaml.org,2002:int";
100                 value = data.toString();
101             } else {
102                 Number number = (Number) data;
103                 tag = "tag:yaml.org,2002:float";
104                 if (number.equals(Double.NaN)) {
105                     value = ".NaN";
106                 } else if (number.equals(Double.POSITIVE_INFINITY)) {
107                     value = ".inf";
108                 } else if (number.equals(Double.NEGATIVE_INFINITY)) {
109                     value = "-.inf";
110                 } else {
111                     value = number.toString();
112                 }
113             }
114             return representScalar(tag, value);
115         }
116     }

141         @SuppressWarnings("unchecked")
142         public Node representData(Object data) {
143             Map<Object, Object> value = new LinkedHashMap<Object, Object>();
144             Set<Object> set = (Set<Object>) data;

152     private class RepresentDate implements Represent {
153         public Node representData(Object data) {
154             // because SimpleDateFormat ignores timezone we have to use Calendar
155             Calendar calendar = Calendar.getInstance(TimeZone.getTimeZone("UTC"));
156             calendar.setTime((Date) data);
157             int years = calendar.get(Calendar.YEAR);
158             int months = calendar.get(Calendar.MONTH) + 1; // 0..12
159             int days = calendar.get(Calendar.DAY_OF_MONTH); // 1..31
160             int hour24 = calendar.get(Calendar.HOUR_OF_DAY); // 0..24
161             int minutes = calendar.get(Calendar.MINUTE); // 0..59
162             int seconds = calendar.get(Calendar.SECOND); // 0..59
163             int millis = calendar.get(Calendar.MILLISECOND);
164             StringBuffer buffer = new StringBuffer(String.valueOf(years));
165             buffer.append("-");
166             if (months < 10) {
167                 buffer.append("0");
168             }
169             buffer.append(String.valueOf(months));
170             buffer.append("-");
171             if (days < 10) {
172                 buffer.append("0");
173             }
174             buffer.append(String.valueOf(days));
175             buffer.append("T");
176             if (hour24 < 10) {
177                 buffer.append("0");
178             }
179             buffer.append(String.valueOf(hour24));
180             buffer.append(":");
181             if (minutes < 10) {
182                 buffer.append("0");
183             }
184             buffer.append(String.valueOf(minutes));
185             buffer.append(":");
186             if (seconds < 10) {
187                 buffer.append("0");
188             }
189             buffer.append(String.valueOf(seconds));
190             if (millis > 0) {
191                 if (millis < 10) {
192                     buffer.append(".00");
193                 } else if (millis < 100) {
194                     buffer.append(".0");
195                 } else {
196                     buffer.append(".");
197                 }
198                 buffer.append(String.valueOf(millis));
199             }
200             buffer.append("Z");
201             return representScalar("tag:yaml.org,2002:timestamp", buffer.toString(), null);
202         }
203     }
