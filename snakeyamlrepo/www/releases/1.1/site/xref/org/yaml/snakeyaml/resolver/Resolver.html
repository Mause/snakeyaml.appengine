17  public class Resolver {
18      private static final String DEFAULT_SCALAR_TAG = "tag:yaml.org,2002:str";
19      private static final String DEFAULT_SEQUENCE_TAG = "tag:yaml.org,2002:seq";
20      private static final String DEFAULT_MAPPING_TAG = "tag:yaml.org,2002:map";
21      private static final Pattern BOOL = Pattern
22              .compile("^(?:yes|Yes|YES|no|No|NO|true|True|TRUE|false|False|FALSE|on|On|ON|off|Off|OFF)$");
23      private static final Pattern FLOAT = Pattern
24              .compile("^(?:[-+]?(?:[0-9][0-9_]*)//.[0-9_]*(?:[eE][-+][0-9]+)?|[-+]?(?:[0-9][0-9_]*)?//.[0-9_]+(?:[eE][-+][0-9]+)?|[-+]?[0-9][0-9_]*(?::[0-5]?[0-9])+//.[0-9_]*|[-+]?//.(?:inf|Inf|INF)|//.(?:nan|NaN|NAN))$");
25      private static final Pattern INT = Pattern
26              .compile("^(?:[-+]?0b[0-1_]+|[-+]?0[0-7_]+|[-+]?(?:0|[1-9][0-9_]*)|[-+]?0x[0-9a-fA-F_]+|[-+]?[1-9][0-9_]*(?::[0-5]?[0-9])+)$");
27      private static final Pattern MERGE = Pattern.compile("^(?:<<)$");
28      private static final Pattern NULL = Pattern.compile("^(?:~|null|Null|NULL| )$");
29      private static final Pattern EMPTY = Pattern.compile("^$");
30      private static final Pattern TIMESTAMP = Pattern
31              .compile("^(?:[0-9][0-9][0-9][0-9]-[0-9][0-9]-[0-9][0-9]|[0-9][0-9][0-9][0-9]-[0-9][0-9]?-[0-9][0-9]?(?:[Tt]|[ \t]+)[0-9][0-9]?:[0-9][0-9]:[0-9][0-9](?://.[0-9]*)?(?:[ \t]*(?:Z|[-+][0-9][0-9]?(?::[0-9][0-9])?))?)$");
32      private static final Pattern VALUE = Pattern.compile("^(?:=)$");
33      private static final Pattern YAML = Pattern.compile("^(?:!|&|//*)$");
34  
35      private Map<Character, List<ResolverTuple>> yamlImplicitResolvers = new HashMap<Character, List<ResolverTuple>>();
36  
37      public Resolver() {
38          addImplicitResolver("tag:yaml.org,2002:bool", BOOL, "yYnNtTfFoO");
39          addImplicitResolver("tag:yaml.org,2002:float", FLOAT, "-+0123456789.");
40          addImplicitResolver("tag:yaml.org,2002:int", INT, "-+0123456789");
41          addImplicitResolver("tag:yaml.org,2002:merge", MERGE, "<");
42          addImplicitResolver("tag:yaml.org,2002:null", NULL, "~nN\0");
43          addImplicitResolver("tag:yaml.org,2002:null", EMPTY, null);
44          addImplicitResolver("tag:yaml.org,2002:timestamp", TIMESTAMP, "0123456789");
45          addImplicitResolver("tag:yaml.org,2002:value", VALUE, "=");
46          // The following implicit resolver is only for documentation purposes.
47          // It cannot work
48          // because plain scalars cannot start with '!', '&', or '*'.
49          addImplicitResolver("tag:yaml.org,2002:yaml", YAML, "!&*");
50      }

52      public void addImplicitResolver(String tag, Pattern regexp, String first) {
53          if (first == null) {
54              List<ResolverTuple> curr = yamlImplicitResolvers.get(null);
55              if (curr == null) {
56                  curr = new LinkedList<ResolverTuple>();
57                  yamlImplicitResolvers.put(null, curr);
58              }
59              curr.add(new ResolverTuple(tag, regexp));
60          } else {
61              char[] chrs = first.toCharArray();
62              for (int i = 0, j = chrs.length; i < j; i++) {
63                  Character theC = new Character(chrs[i]);
64                  if (theC == 0) {
65                      // special case: for null
66                      theC = null;
67                  }
68                  List<ResolverTuple> curr = yamlImplicitResolvers.get(theC);
69                  if (curr == null) {
70                      curr = new LinkedList<ResolverTuple>();
71                      yamlImplicitResolvers.put(theC, curr);
72                  }
73                  curr.add(new ResolverTuple(tag, regexp));
74              }
75          }
76      }

78      public String resolve(NodeId kind, String value, boolean implicit) {
79          if (kind == NodeId.scalar && implicit) {
80              List<ResolverTuple> resolvers = null;
81              if ("".equals(value)) {
82                  resolvers = yamlImplicitResolvers.get('\0');
83              } else {
84                  resolvers = yamlImplicitResolvers.get(value.charAt(0));
85              }
86              if (resolvers != null) {
87                  for (ResolverTuple v : resolvers) {
88                      String tag = v.getTag();
89                      Pattern regexp = v.getRegexp();
90                      if (regexp.matcher(value).matches()) {
91                          return tag;
92                      }
93                  }
94              }
95              if (yamlImplicitResolvers.containsKey(null)) {
96                  for (ResolverTuple v : yamlImplicitResolvers.get(null)) {
97                      String tag = v.getTag();
98                      Pattern regexp = v.getRegexp();
99                      if (regexp.matcher(value).matches()) {
100                         return tag;
101                     }
102                 }
103             }
104         }
105         switch (kind) {
106         case scalar:
107             return DEFAULT_SCALAR_TAG;
108         case sequence:
109             return DEFAULT_SEQUENCE_TAG;
110         default:
111             return DEFAULT_MAPPING_TAG;
112         }
113     }
