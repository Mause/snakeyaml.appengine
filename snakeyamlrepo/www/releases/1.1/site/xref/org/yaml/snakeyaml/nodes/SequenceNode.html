13  public class SequenceNode extends CollectionNode {
14      private Class<? extends Object> listType;
15  
16      public SequenceNode(String tag, List<Node> value, Mark startMark, Mark endMark,
17              Boolean flowStyle) {
18          super(tag, value, startMark, endMark, flowStyle);
19          listType = Object.class;
20      }
21  
22      public SequenceNode(String tag, List<Node> value, Boolean flowStyle) {
23          this(tag, value, null, null, flowStyle);
24      }
25  
26      @Override
27      public NodeId getNodeId() {
28          return NodeId.sequence;
29      }
30  
31      @SuppressWarnings("unchecked")
32      public List<Node> getValue() {
33          List<Node> value = (List<Node>) super.getValue();
34          for (Node node : value) {
35              node.setType(listType);
36          }
37          return value;
38      }

40      public void setListType(Class<? extends Object> listType) {
41          this.listType = listType;
42      }
