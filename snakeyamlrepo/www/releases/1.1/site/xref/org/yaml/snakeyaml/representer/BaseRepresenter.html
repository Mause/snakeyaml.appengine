21  public abstract class BaseRepresenter {
22      @SuppressWarnings("unchecked")
23      protected final Map<Class, Represent> representers = new HashMap<Class, Represent>();
24      /***
25       * in Java 'null' is not a type. So we have to keep the null representer
26       * separately otherwise it will coincide with the default representer which
27       * is stored with the key null.
28       */
29      protected Represent nullRepresenter;
30      @SuppressWarnings("unchecked")
31      protected final Map<Class, Represent> multiRepresenters = new HashMap<Class, Represent>();
32      private Character defaultStyle;
33      protected Boolean defaultFlowStyle;
34      protected final Map<Integer, Node> representedObjects = new HashMap<Integer, Node>();
35      private final Set<Object> objectKeeper = new HashSet<Object>();
36      protected Integer aliasKey;// internal memory address
37      protected String rootTag = null;
38  
39      public BaseRepresenter(Character default_style, Boolean default_flow_style) {
40          this.defaultStyle = default_style;
41          this.defaultFlowStyle = default_flow_style;
42      }
43  
44      public void represent(Serializer serializer, Object data) throws IOException {
45          Node node = representData(data);
46          serializer.serialize(node);
47          representedObjects.clear();
48          objectKeeper.clear();
49      }
50  
51      @SuppressWarnings("unchecked")
52      protected Node representData(Object data) {
53          aliasKey = System.identityHashCode(data);// take memory address
54          if (!ignoreAliases(data)) {
55              // check for identity
56              if (representedObjects.containsKey(aliasKey)) {
57                  Node node = representedObjects.get(aliasKey);
58                  return node;
59              }
60          }
61          // check for null first
62          if (data == null) {
63              Node node = nullRepresenter.representData(data);
64              return node;
65          }
66          // check the same class
67          Node node;
68          Class clazz = data.getClass();
69          if (representers.containsKey(clazz)) {
70              Represent representer = representers.get(clazz);
71              node = representer.representData(data);
72          } else {
73              // check the parents
74              for (Class repr : multiRepresenters.keySet()) {
75                  if (repr.isInstance(data)) {
76                      Represent representer = multiRepresenters.get(repr);
77                      node = representer.representData(data);
78                      return node;
79                  }
80              }
81              // check array of primitives
82              if (clazz.isArray()) {
83                  throw new YAMLException("Arrays of primitives are not fully supported.");
84              }
85              // check defaults
86              if (multiRepresenters.containsKey(null)) {
87                  Represent representer = multiRepresenters.get(null);
88                  node = representer.representData(data);
89              } else {
90                  Represent representer = representers.get(null);
91                  node = representer.representData(data);
92              }
93          }
94          return node;
95      }
96  
97      protected Node representScalar(String tag, String value, Character style) {
98          if (style == null) {
99              style = this.defaultStyle;
100         }
101         Node node = new ScalarNode(tag, value, null, null, style);
102         representedObjects.put(aliasKey, node);
103         return node;
104     }
105 
106     protected Node representScalar(String tag, String value) {
107         return representScalar(tag, value, null);
108     }
109 
110     protected Node representSequence(String tag, List<? extends Object> sequence, Boolean flowStyle) {
111         List<Node> value = new LinkedList<Node>();
112         SequenceNode node = new SequenceNode(tag, value, flowStyle);
113         representedObjects.put(aliasKey, node);
114         boolean bestStyle = true;
115         for (Object item : sequence) {
116             Node nodeItem = representData(item);
117             if (!((nodeItem instanceof ScalarNode && ((ScalarNode) nodeItem).getStyle() == null))) {
118                 bestStyle = false;
119             }
120             value.add(nodeItem);
121         }
122         if (flowStyle == null) {
123             if (defaultFlowStyle != null) {
124                 node.setFlowStyle(defaultFlowStyle);
125             } else {
126                 node.setFlowStyle(bestStyle);
127             }
128         }
129         return node;
130     }
131 
132     protected Node representMapping(String tag, Map<? extends Object, Object> mapping,
133             Boolean flowStyle) {
134         List<Node[]> value = new LinkedList<Node[]>();
135         MappingNode node = new MappingNode(tag, value, flowStyle);
136         representedObjects.put(aliasKey, node);
137         boolean bestStyle = true;
138         for (Object itemKey : mapping.keySet()) {
139             Object itemValue = mapping.get(itemKey);
140             Node nodeKey = representData(itemKey);
141             Node nodeValue = representData(itemValue);
142             if (!((nodeKey instanceof ScalarNode && ((ScalarNode) nodeKey).getStyle() == null))) {
143                 bestStyle = false;
144             }
145             if (!((nodeValue instanceof ScalarNode && ((ScalarNode) nodeValue).getStyle() == null))) {
146                 bestStyle = false;
147             }
148             value.add(new Node[] { nodeKey, nodeValue });
149         }
150         if (flowStyle == null) {
151             if (defaultFlowStyle != null) {
152                 node.setFlowStyle(defaultFlowStyle);
153             } else {
154                 node.setFlowStyle(bestStyle);
155             }
156         }
157         return node;
158     }

160     protected abstract boolean ignoreAliases(Object data);
