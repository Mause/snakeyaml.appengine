7   import java.nio.charset.Charset;
8   import java.util.regex.Matcher;
9   import java.util.regex.Pattern;

11  import org.yaml.snakeyaml.error.Mark;
12  import org.yaml.snakeyaml.error.YAMLException;

14  /***
15   * Reader: determines the data encoding and converts it to unicode, checks if
16   * characters are in allowed range, adds '\0' to the end.
17   * 
18   * @see <a href="http://pyyaml.org/wiki/PyYAML">PyYAML</a> for more information
19   */
20  public class Reader {
21      // NON_PRINTABLE changed from PyYAML: \uFFFD excluded because Java returns
22      // it in case of data corruption
23      final static Pattern NON_PRINTABLE = Pattern
24              .compile("[^\t\n\r\u0020-\u007E\u0085\u00A0-\uD7FF\uE000-\uFFFC]");
25      private final static String LINEBR = "\n\u0085\u2028\u2029";
26  
27      private String name;
28      private final java.io.Reader stream;
29      private int pointer = 0;
30      private boolean eof = true;
31      private final StringBuffer buffer;
32      private int index = 0;
33      private int line = 0;
34      private int column = 0;
35  
36      public Reader(String stream) {
37          this.name = "<string>";
38          this.buffer = new StringBuffer();
39          checkPrintable(stream);
40          this.buffer.append(stream);
41          this.stream = null;
42          this.eof = true;
43      }
44  
45      public Reader(java.io.Reader reader) {
46          this.name = "<reader>";
47          this.buffer = new StringBuffer();
48          this.stream = reader;
49          this.eof = false;
50      }
51  
52      void checkPrintable(CharSequence data) {
53          Matcher em = NON_PRINTABLE.matcher(data);
54          if (em.find()) {
55              int position = this.index + this.buffer.length() - this.pointer + em.start();
56              throw new ReaderException(name, position, em.group().charAt(0),
57                      " special characters are not allowed");
58          }
59      }
60  
61      public Mark getMark() {
62          if (this.stream == null) {
63              return new Mark(name, this.index, this.line, this.column, this.buffer.toString(),
64                      this.pointer);
65          } else {
66              return new Mark(name, this.index, this.line, this.column, null, 0);
67          }
68      }
69  
70      public void forward() {
71          forward(1);
72      }
73  
74      /***
75       * read the next length characters and move the pointer.
76       * 
77       * @param length
78       */
79      public void forward(int length) {
80          if (this.pointer + length + 1 >= this.buffer.length()) {
81              update(length + 1);
82          }
83          char ch = 0;
84          for (int i = 0; i < length; i++) {
85              ch = this.buffer.charAt(this.pointer);
86              this.pointer++;
87              this.index++;
88              if (LINEBR.indexOf(ch) != -1
89                      || (ch == '\r' && this.buffer.charAt(this.pointer) != '\n')) {
90                  this.line++;
91                  this.column = 0;
92              } else if (ch != '\uFEFF') {
93                  this.column++;
94              }
95          }
96      }

98      public char peek() {
99          return peek(0);
100     }

102     /***
103      * Peek the next index-th character
104      * 
105      * @param index
106      * @return
107      */
108     public char peek(int index) {
109         if (this.pointer + index + 1 > this.buffer.length()) {
110             update(index + 1);
111         }
112         return this.buffer.charAt(this.pointer + index);
113     }
114 
115     /***
116      * peek the next length characters
117      * 
118      * @param length
119      * @return
120      */
121     public String prefix(int length) {
122         if (this.pointer + length >= this.buffer.length()) {
123             update(length);
124         }
125         if (this.pointer + length > this.buffer.length()) {
126             return this.buffer.substring(this.pointer, this.buffer.length());
127         } else {
128             return this.buffer.substring(this.pointer, this.pointer + length);
129         }
130     }
131 
132     private void update(int length) {
133         this.buffer.delete(0, this.pointer);
134         this.pointer = 0;
135         while (this.buffer.length() < length) {
136             String rawData = "";
137             if (!this.eof) {
138                 char[] data = new char[1024];
139                 int converted = -2;
140                 try {
141                     converted = this.stream.read(data);
142                 } catch (IOException ioe) {
143                     throw new YAMLException(ioe);
144                 }
145                 if (converted == -1) {
146                     this.eof = true;
147                 } else {
148                     rawData = new String(data, 0, converted);
149                 }
150             }
151             checkPrintable(rawData);
152             this.buffer.append(rawData);
153             if (this.eof) {
154                 this.buffer.append('\0');
155                 break;
156             }
157         }
158     }
