9   import org.yaml.snakeyaml.Loader;
10  import org.yaml.snakeyaml.composer.Composer;

13  public class CanonicalLoader extends Loader {
14      @Override
15      public Object load(java.io.Reader yaml) {
16          try {
17              int ch = yaml.read();
18              StringBuffer buffer = new StringBuffer();
19              while (ch != -1) {
20                  buffer.append((char) ch);
21                  ch = yaml.read();
22              }
23              Composer composer = new Composer(new CanonicalParser(buffer.toString()), resolver);
24              constructor.setComposer(composer);
25              return constructor.getSingleData();
26          } catch (IOException e) {
27              throw new YAMLException(e);
28          }
29      }

31      public Iterable<Object> loadAll(java.io.Reader yaml) {
32          try {
33              int ch = yaml.read();
34              StringBuffer buffer = new StringBuffer();
35              while (ch != -1) {
36                  buffer.append((char) ch);
37                  ch = yaml.read();
38              }
39              Composer composer = new Composer(new CanonicalParser(buffer.toString()), resolver);
40              this.constructor.setComposer(composer);
41              Iterator<Object> result = new Iterator<Object>() {
42                  public boolean hasNext() {
43                      return constructor.checkData();
44                  }
45  
46                  public Object next() {
47                      return constructor.getData();
48                  }
49  
50                  public void remove() {
51                      throw new UnsupportedOperationException();
52                  }
53              };
54              return new YamlIterable(result);
55          } catch (IOException e) {
56              throw new YAMLException(e);
57          }
58      }
59  
60      private class YamlIterable implements Iterable<Object> {
61          private Iterator<Object> iterator;
62  
63          public YamlIterable(Iterator<Object> iterator) {
64              this.iterator = iterator;
65          }
66  
67          public Iterator<Object> iterator() {
68              return iterator;
69          }
70  
71      }
