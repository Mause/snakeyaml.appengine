26      public void testGetEvent() {
27          String data = "string: abcd";
28          Reader reader = new Reader(data);
29          Parser parser = new ParserImpl(reader);
30          Mark dummyMark = new Mark("dummy", 0, 0, 0, "", 0);
31          LinkedList<Event> etalonEvents = new LinkedList<Event>();
32          etalonEvents.add(new StreamStartEvent(dummyMark, dummyMark));
33          etalonEvents.add(new DocumentStartEvent(dummyMark, dummyMark, false, null, null));
34          etalonEvents.add(new MappingStartEvent(null, null, true, dummyMark, dummyMark,
35                  Boolean.FALSE));
36          boolean[] implicit = { true, false };
37          etalonEvents.add(new ScalarEvent(null, null, implicit, "string", dummyMark, dummyMark,
38                  (char) 0));
39          etalonEvents.add(new ScalarEvent(null, null, implicit, "abcd", dummyMark, dummyMark,
40                  (char) 0));
41          etalonEvents.add(new MappingEndEvent(dummyMark, dummyMark));
42          etalonEvents.add(new DocumentEndEvent(dummyMark, dummyMark, false));
43          etalonEvents.add(new StreamEndEvent(dummyMark, dummyMark));
44          while (parser.checkEvent(new ArrayList<Class<? extends Event>>())) {
45              Event event = parser.getEvent();
46              if (etalonEvents.isEmpty()) {
47                  fail("unexpected event: " + event);
48              }
49              assertEquals(etalonEvents.removeFirst(), event);
50          }
51          assertFalse("Must contain no more events: " + parser.getEvent(), parser
52                  .checkEvent(new ArrayList<Class<? extends Event>>()));
53      }
