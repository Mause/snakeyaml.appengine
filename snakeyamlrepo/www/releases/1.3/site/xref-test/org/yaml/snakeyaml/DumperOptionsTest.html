16      public void testDefaultStyle() {
17          DumperOptions options = new DumperOptions();
18          Yaml yaml = new Yaml(options);
19          assertEquals("abc\n", yaml.dump("abc"));
20          // string which looks like integer
21          assertEquals("'123'\n", yaml.dump("123"));
22          //
23          options.setDefaultScalarStyle(DumperOptions.ScalarStyle.DOUBLE_QUOTED);
24          yaml = new Yaml(options);
25          assertEquals("\"123\"\n", yaml.dump("123"));
26          //
27          options.setDefaultScalarStyle(DumperOptions.ScalarStyle.SINGLE_QUOTED);
28          yaml = new Yaml(options);
29          assertEquals("'123'\n", yaml.dump("123"));
30          //
31          options.setDefaultScalarStyle(DumperOptions.ScalarStyle.PLAIN);
32          yaml = new Yaml(options);
33          assertEquals("'123'\n", yaml.dump("123"));

35          // null check
36          try {
37              options.setDefaultScalarStyle(null);

39          } catch (NullPointerException e) {
40              assertEquals("Use ScalarStyle enum.", e.getMessage());
