15  public class ComposerImplTest extends TestCase {

17      public void testGetNode() {
18          String data = "american:\n  - Boston Red Sox";
19          Yaml yaml = new Yaml();
20          Node node = yaml.compose(new StringReader(data));
21          assertNotNull(node);
22          assertTrue(node instanceof MappingNode);
23          String data2 = "---\namerican:\n- Boston Red Sox";
24          Node node2 = yaml.compose(new StringReader(data2));
25          assertNotNull(node2);
26          assertFalse(node.equals(node2));
27      }

29      public void testComposeBean() {
30          String data = "!!org.yaml.snakeyaml.composer.ComposerImplTest$BeanToCompose {name: Bill, age: 18}";
31          Yaml yaml = new Yaml();
32          Node node = yaml.compose(new StringReader(data));
33          assertNotNull(node);
34          assertTrue(node instanceof MappingNode);
35          assertEquals(
36                  "tag:yaml.org,2002:org.yaml.snakeyaml.composer.ComposerImplTest$BeanToCompose",
37                  node.getTag());
38          assertEquals(NodeId.mapping, node.getNodeId());
39          assertEquals(Object.class, node.getType());
40      }
