18      public void testTimestamp() throws IOException {
19          assertEquals("2001-12-15 at 2:59:43 (100)", getText("canonical: 2001-12-15T02:59:43.1Z",
20                  "canonical"));
21          // zero miliseconds
22          assertEquals("2001-12-15 at 2:59:43 (0)", getText("canonical: 2001-12-15T02:59:43.000Z",
23                  "canonical"));
24          assertEquals("2001-12-15 at 2:59:43 (100)", getText(
25                  "valid iso8601:    2001-12-14t21:59:43.10-05:00", "valid iso8601"));
26          // half hour time zone
27          assertEquals("2001-12-14 at 22:29:43 (100)", getText(
28                  "valid iso8601:    2001-12-14t21:59:43.10-0:30", "valid iso8601"));
29          // + time zone
30          assertEquals("2001-12-14 at 19:59:43 (100)", getText(
31                  "valid iso8601:    2001-12-14t21:59:43.10+2:00", "valid iso8601"));
32          assertEquals("2001-12-15 at 2:59:43 (100)", getText(
33                  "space separated:  2001-12-14 21:59:43.10 -5", "space separated"));
34          assertEquals("2001-12-15 at 2:59:43 (100)", getText(
35                  "no time zone (Z): 2001-12-15 2:59:43.10", "no time zone (Z)"));
36          assertEquals("2002-12-14 at 0:0:0 (0)", getText("date (00:00:00Z): 2002-12-14",
37                  "date (00:00:00Z)"));
38      }

69      private String getText(String yaml, String key) {
70          Date date = (Date) getMapValue(yaml, key);
71          Calendar cal = Calendar.getInstance(TimeZone.getTimeZone("UTC"));
72          cal.setTime(date);
73          int years = cal.get(Calendar.YEAR);
74          int months = cal.get(Calendar.MONTH) + 1; // 0..12
75          int days = cal.get(Calendar.DAY_OF_MONTH); // 1..31
76          int hour24 = cal.get(Calendar.HOUR_OF_DAY); // 0..24
77          int minutes = cal.get(Calendar.MINUTE); // 0..59
78          int seconds = cal.get(Calendar.SECOND); // 0..59
79          int millis = cal.get(Calendar.MILLISECOND);
80          String result = String.valueOf(years) + "-" + String.valueOf(months) + "-"
81                  + String.valueOf(days) + " at " + String.valueOf(hour24) + ":"
82                  + String.valueOf(minutes) + ":" + String.valueOf(seconds) + " ("
83                  + String.valueOf(millis) + ")";
84          return result;
85      }
