4   package org.yaml.snakeyaml;

6   import java.sql.Date;
7   import java.sql.Timestamp;

9   import junit.framework.TestCase;
10  
11  import org.yaml.snakeyaml.DumperOptions.FlowStyle;
12  
13  public class JavaBeanTimeStampTest extends TestCase {
14      public void testLoadDefaultJavaSqlTimestamp() throws Exception {
15          JavaBeanWithSqlTimestamp javaBeanToDump = new JavaBeanWithSqlTimestamp();
16          Timestamp stamp = new Timestamp(1000000000000L);
17          javaBeanToDump.setTimestamp(stamp);
18          Date date = new Date(1001376000000L);
19          javaBeanToDump.setDate(date);
20          DumperOptions options = new DumperOptions();
21          options.setDefaultFlowStyle(FlowStyle.BLOCK);
22          Yaml yaml = new Yaml(options);
23          String dumpStr = yaml.dump(javaBeanToDump);
24          assertEquals(
25                  "!!org.yaml.snakeyaml.JavaBeanWithSqlTimestamp\ndate: 2001-09-25T00:00:00Z\ntimestamp: 2001-09-09T01:46:40Z\n",
26                  dumpStr);
27          JavaBeanWithSqlTimestamp javaBeanToLoad = JavaBeanParser.load(dumpStr,
28                  JavaBeanWithSqlTimestamp.class);
29          assertEquals(stamp, javaBeanToLoad.getTimestamp());
30          assertEquals(date, javaBeanToLoad.getDate());
31      }
32  }
