4   package org.yaml.snakeyaml.constructor;

6   import java.io.IOException;
7   import java.math.BigDecimal;

9   import junit.framework.TestCase;
10  
11  import org.yaml.snakeyaml.Yaml;
12  
13  public class BigDecimalBeanConstructorTest extends TestCase {
14  
15      public void testRepresentor() throws IOException {
16          BigDecimalJavaBean bean = new BigDecimalJavaBean();
17          bean.setAmount(1.5f);
18          bean.setNumber(new BigDecimal("3.1416"));
19          Yaml yaml = new Yaml();
20          String output = yaml.dump(bean);
21          String className = this.getClass().getPackage().getName();
22          assertEquals("!!" + className + ".BigDecimalJavaBean {amount: 1.5, number: 3.1416}\n",
23                  output);
24      }
25  
26      public void testConstructor() throws IOException {
27          String className = "!!" + this.getClass().getPackage().getName()
28                  + ".BigDecimalJavaBean {amount: 1.5, number: 3.1416}";
29          Yaml yaml = new Yaml();
30          BigDecimalJavaBean bean = (BigDecimalJavaBean) yaml.load(className);
31          assertNotNull(bean);
32          assertTrue(1.5 - bean.getAmount() < 0.0000001);
33          assertTrue((new BigDecimal("3.1416")).add(bean.getNumber().negate()).doubleValue() < 0.0000001);
34      }
35  
36      public void testConstructorAtomic() throws IOException {
37          String className = "!!" + this.getClass().getPackage().getName()
38                  + ".AtomicJavaBean {amount: 1.5, atomic: 0}";
39          Yaml yaml = new Yaml();
40          try {
41              yaml.load(className);
42              fail("AtomicLong is not supported.");
43          } catch (Exception e) {
44              assertEquals(
45                      "org.yaml.snakeyaml.error.YAMLException: Unsupported class: class java.util.concurrent.atomic.AtomicLong",
46                      e.getCause().getMessage());
47          }
48      }
