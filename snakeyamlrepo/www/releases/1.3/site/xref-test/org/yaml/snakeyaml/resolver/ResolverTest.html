28      @SuppressWarnings("unchecked")
29      public void testAddImplicitResolver() {
30          Dumper dumper = new Dumper(new MyRepresenter(), new DumperOptions());
31          Loader loader = new Loader(new MyConstructor());
32          Yaml yaml = new Yaml(loader, dumper);
33          Pattern regexp = Pattern.compile("//d//d-//d//d-//d//d//d");
34          yaml.addImplicitResolver("tag:yaml.org,2002:Phone", regexp, "0123456789");
35          Phone phone1 = new Phone("12-34-567");
36          Phone phone2 = new Phone("11-22-333");
37          Phone phone3 = new Phone("44-55-777");
38          List<Phone> etalonList = new LinkedList<Phone>();
39          etalonList.add(phone1);
40          etalonList.add(phone2);
41          etalonList.add(phone3);
42          String output = yaml.dump(etalonList);
43          assertEquals("[12-34-567, 11-22-333, 44-55-777]\n", output);
44          List<Phone> parsedList = (List<Phone>) yaml.load(output);
45          assertEquals(3, parsedList.size());
46          assertEquals(phone1, parsedList.get(0));
47          assertEquals(phone2, parsedList.get(1));
48          assertEquals(phone3, parsedList.get(2));
49          assertEquals(etalonList, parsedList);
50      }

52      public void testAddImplicitResolver2() {
53          Dumper dumper = new Dumper(new PointRepresenter(), new DumperOptions());
54          Yaml yaml = new Yaml(dumper);
55          Pattern regexp = Pattern.compile("//d//d-//d//d-//d//d//d");
56          yaml.addImplicitResolver("tag:yaml.org,2002:Phone", regexp, "\0");
57          Pattern regexp2 = Pattern.compile("x//d_y//d");
58          // try any scalar, and not only those which start with 'x'
59          yaml.addImplicitResolver("tag:yaml.org,2002:Point", regexp2, null);
60          Map<String, Object> map = new LinkedHashMap<String, Object>();
61          map.put("a", new Phone("12-34-567"));
62          map.put("b", new Point(1, 5));
63          String output = yaml.dump(map);
64          assertEquals("{a: 12-34-567, b: x1_y5}\n", output);
65      }
