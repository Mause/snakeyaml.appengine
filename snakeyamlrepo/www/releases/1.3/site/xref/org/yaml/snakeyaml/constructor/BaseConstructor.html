25  public abstract class BaseConstructor {
26      protected final Map<String, Construct> yamlConstructors = new HashMap<String, Construct>();
27  
28      private Composer composer;
29      private final Map<Node, Object> constructedObjects;
30      private final Set<Node> recursiveObjects;
31      private final Stack<Tuple<Node, Object>> toBeConstructedAt2ndStep;
32      private final LinkedList<Tuple<Map<Object, Object>, Tuple<Object, Object>>> maps2fill;
33      private final LinkedList<Tuple<Set<Object>, Object>> sets2fill;
34  
35      protected Class<? extends Object> rootType;
36  
37      public BaseConstructor() {
38          constructedObjects = new HashMap<Node, Object>();
39          recursiveObjects = new HashSet<Node>();
40          toBeConstructedAt2ndStep = new Stack<Tuple<Node, Object>>();
41          maps2fill = new LinkedList<Tuple<Map<Object, Object>, Tuple<Object, Object>>>();
42          sets2fill = new LinkedList<Tuple<Set<Object>, Object>>();
43          rootType = Object.class;
44      }

46      public void setComposer(Composer composer) {
47          this.composer = composer;
48      }

50      public boolean checkData() {
51          // If there are more documents available?
52          return composer.checkNode();
53      }
54  
55      public Object getData() {
56          // Construct and return the next document.
57          composer.checkNode();
58          Node node = composer.getNode();
59          node.setType(rootType);
60          return constructDocument(node);
61      }
62  
63      public Object getSingleData() {
64          // Ensure that the stream contains a single document and construct it
65          Node node = composer.getSingleNode();
66          if (node != null) {
67              node.setType(rootType);
68              return constructDocument(node);
69          }
70          return null;
71      }

73      private Object constructDocument(Node node) {
74          Object data = constructObject(node);
75          while (!toBeConstructedAt2ndStep.isEmpty()) {
76              Tuple<Node, Object> toBeProcessed = toBeConstructedAt2ndStep.pop();
77              callPostCreate(toBeProcessed._1(), toBeProcessed._2());
78          }
79          if (!maps2fill.isEmpty()) {
80              for (Tuple<Map<Object, Object>, Tuple<Object, Object>> entry : maps2fill) {
81                  Tuple<Object, Object> key_value = entry._2();
82                  entry._1().put(key_value._1(), key_value._2());
83              }
84              maps2fill.clear();
85          }
86          if (!sets2fill.isEmpty()) {
87              for (Tuple<Set<Object>, Object> value : sets2fill) {
88                  value._1().add(value._2());
89              }
90              sets2fill.clear();
91          }
92          constructedObjects.clear();
93          recursiveObjects.clear();
94          toBeConstructedAt2ndStep.clear();
95          return data;
96      }
97  
98      protected Object constructObject(Node node) {
99          if (constructedObjects.containsKey(node)) {
100             return constructedObjects.get(node);
101         }
102         if (recursiveObjects.contains(node)) {
103             throw new ConstructorException(null, null, "found unconstructable recursive node", node
104                     .getStartMark());
105         }
106         recursiveObjects.add(node);
107         Object data = callConstructor(node);
108         if (node.isTwoStepsConstruction()) {
109             toBeConstructedAt2ndStep.push(new Tuple<Node, Object>(node, data));
110         }
111         constructedObjects.put(node, data);
112         recursiveObjects.remove(node);
113         return data;
114     }

124     private Construct getConstructor(Node node) {
125         Construct constructor = yamlConstructors.get(node.getTag());
126         if (constructor == null) {

174     protected void constructMapping2ndStep(MappingNode node, Map<Object, Object> mapping) {
175         List<Node[]> nodeValue = (List<Node[]>) node.getValue();
176         for (Node[] tuple : nodeValue) {
177             Node keyNode = tuple[0];
178             Node valueNode = tuple[1];
179             Object key = constructObject(keyNode);
180             if (key != null) {
181                 try {
182                     key.hashCode();// check circular dependencies
183                 } catch (Exception e) {
184                     throw new ConstructorException("while constructing a mapping", node
185                             .getStartMark(), "found unacceptable key " + key, tuple[0]
186                             .getStartMark(), e);
187                 }
188             }
189             Object value = constructObject(valueNode);
190             if (keyNode.isTwoStepsConstruction()) {
191                 /*
192                  * if keyObject is created it 2 steps we should postpone putting
193                  * it in map because it may have different hash after
194                  * initialization compared to clean just created one. And map of
195                  * course does not observe key hashCode changes.
196                  */
197                 maps2fill.addFirst(new Tuple<Map<Object, Object>, Tuple<Object, Object>>(mapping,
198                         new Tuple<Object, Object>(key, value)));
199             } else {
200                 mapping.put(key, value);
201             }
202         }
203     }
204 
205     protected void constructSet2ndStep(MappingNode node, Set<Object> set) {
206         List<Node[]> nodeValue = (List<Node[]>) node.getValue();
207         for (Node[] tuple : nodeValue) {
208             Node keyNode = tuple[0];
209             Object key = constructObject(keyNode);
210             if (key != null) {
211                 try {
212                     key.hashCode();// check circular dependencies
213                 } catch (Exception e) {
214                     throw new ConstructorException("while constructing a Set", node.getStartMark(),
215                             "found unacceptable key " + key, tuple[0].getStartMark(), e);
216                 }
217             }
218             if (keyNode.isTwoStepsConstruction()) {
219                 /*
220                  * if keyObject is created it 2 steps we should postpone putting
221                  * it into the set because it may have different hash after
222                  * initialization compared to clean just created one. And set of
223                  * course does not observe value hashCode changes.
224                  */
225                 sets2fill.addFirst(new Tuple<Set<Object>, Object>(set, key));
226             } else {
227                 set.add(key);
228             }
229         }
230     }
