49      private void flattenMapping(MappingNode node) {
50          List<Node[]> merge = new LinkedList<Node[]>();
51          int index = 0;
52          List<Node[]> nodeValue = (List<Node[]>) node.getValue();
53          while (index < nodeValue.size()) {
54              Node keyNode = nodeValue.get(index)[0];
55              Node valueNode = nodeValue.get(index)[1];
56              if (keyNode.getTag().equals("tag:yaml.org,2002:merge")) {
57                  nodeValue.remove(index);
58                  switch (valueNode.getNodeId()) {
59                  case mapping:
60                      MappingNode mn = (MappingNode) valueNode;
61                      flattenMapping(mn);
62                      merge.addAll(mn.getValue());
63                      break;
64                  case sequence:
65                      List<List<Node[]>> submerge = new LinkedList<List<Node[]>>();
66                      SequenceNode sn = (SequenceNode) valueNode;
67                      List<Node> vals = sn.getValue();
68                      for (Node subnode : vals) {
69                          if (!(subnode instanceof MappingNode)) {
70                              throw new ConstructorException("while constructing a mapping", node
71                                      .getStartMark(), "expected a mapping for merging, but found "
72                                      + subnode.getNodeId(), subnode.getStartMark());
73                          }
74                          MappingNode mnode = (MappingNode) subnode;
75                          flattenMapping(mnode);
76                          submerge.add(mnode.getValue());
77                      }
78                      Collections.reverse(submerge);
79                      for (List<Node[]> value : submerge) {
80                          merge.addAll(value);
81                      }
82                      break;
83                  default:
84                      throw new ConstructorException("while constructing a mapping", node
85                              .getStartMark(),
86                              "expected a mapping or list of mappings for merging, but found "
87                                      + valueNode.getNodeId(), valueNode.getStartMark());
88                  }
89              } else if (keyNode.getTag().equals("tag:yaml.org,2002:value")) {
90                  keyNode.setTag("tag:yaml.org,2002:str");
91                  index++;
92              } else {
93                  index++;
94              }
95          }
96          if (!merge.isEmpty()) {
97              merge.addAll(nodeValue);
98              ((MappingNode) node).setValue(merge);
99          }
100     }

137     private class ConstructYamlInt extends AbstractConstruct {
138         public Object construct(Node node) {
139             String value = constructScalar((ScalarNode) node).toString().replaceAll("_", "");
140             int sign = +1;
141             char first = value.charAt(0);
142             if (first == '-') {
143                 sign = -1;
144                 value = value.substring(1);
145             } else if (first == '+') {
146                 value = value.substring(1);
147             }
148             int base = 10;
149             if ("0".equals(value)) {
150                 return new Integer(0);
151             } else if (value.startsWith("0b")) {
152                 value = value.substring(2);
153                 base = 2;
154             } else if (value.startsWith("0x")) {
155                 value = value.substring(2);
156                 base = 16;
157             } else if (value.startsWith("0")) {
158                 value = value.substring(1);
159                 base = 8;
160             } else if (value.indexOf(':') != -1) {
161                 String[] digits = value.split(":");
162                 int bes = 1;
163                 int val = 0;
164                 for (int i = 0, j = digits.length; i < j; i++) {
165                     val += (Long.parseLong(digits[(j - i) - 1]) * bes);
166                     bes *= 60;
167                 }
168                 return createNumber(sign, String.valueOf(val), 10);
169             } else {
170                 return createNumber(sign, value, 10);
171             }
172             return createNumber(sign, value, base);
173         }
174     }

176     private Number createNumber(int sign, String number, int radix) {
177         Number result;
178         if (sign < 0) {
179             number = "-" + number;
180         }
181         try {
182             int integer = Integer.parseInt(number, radix);
183             result = new Integer(integer);
184         } catch (NumberFormatException e) {
185             try {
186                 long longValue = Long.parseLong(number, radix);
187                 result = new Long(longValue);
188             } catch (NumberFormatException e1) {
189                 result = new BigInteger(number, radix);
190             }
191         }
192         return result;
193     }

195     private class ConstructYamlFloat extends AbstractConstruct {
196         public Object construct(Node node) {
197             String value = constructScalar((ScalarNode) node).toString().replaceAll("_", "");
198             int sign = +1;
199             char first = value.charAt(0);
200             if (first == '-') {
201                 sign = -1;
202                 value = value.substring(1);
203             } else if (first == '+') {
204                 value = value.substring(1);
205             }
206             String valLower = value.toLowerCase();
207             if (".inf".equals(valLower)) {
208                 return new Double(sign == -1 ? Double.NEGATIVE_INFINITY : Double.POSITIVE_INFINITY);
209             } else if (".nan".equals(valLower)) {
210                 return new Double(Double.NaN);
211             } else if (value.indexOf(':') != -1) {
212                 String[] digits = value.split(":");
213                 int bes = 1;
214                 double val = 0.0;
215                 for (int i = 0, j = digits.length; i < j; i++) {
216                     val += (Double.parseDouble(digits[(j - i) - 1]) * bes);
217                     bes *= 60;
218                 }
219                 return new Double(sign * val);
220             } else {
221                 Double d = Double.valueOf(value);
222                 return new Double(d.doubleValue() * sign);
223             }
224         }

235     private final static Pattern TIMESTAMP_REGEXP = Pattern
236             .compile("^([0-9][0-9][0-9][0-9])-([0-9][0-9]?)-([0-9][0-9]?)(?:(?:[Tt]|[ \t]+)([0-9][0-9]?):([0-9][0-9]):([0-9][0-9])(?://.([0-9]*))?(?:[ \t]*(?:Z|([-+][0-9][0-9]?)(?::([0-9][0-9])?)?))?)?$");
237     private final static Pattern YMD_REGEXP = Pattern
238             .compile("^([0-9][0-9][0-9][0-9])-([0-9][0-9]?)-([0-9][0-9]?)$");

240     private class ConstructYamlTimestamp extends AbstractConstruct {
241         public Object construct(Node node) {
242             Matcher match = YMD_REGEXP.matcher((String) node.getValue());
243             if (match.matches()) {
244                 String year_s = match.group(1);
245                 String month_s = match.group(2);
246                 String day_s = match.group(3);
247                 Calendar cal = Calendar.getInstance(TimeZone.getTimeZone("UTC"));
248                 cal.clear();
249                 cal.set(Calendar.YEAR, Integer.parseInt(year_s));
250                 // Java's months are zero-based...
251                 cal.set(Calendar.MONTH, Integer.parseInt(month_s) - 1); // x
252                 cal.set(Calendar.DAY_OF_MONTH, Integer.parseInt(day_s));
253                 return cal.getTime();
254             } else {
255                 match = TIMESTAMP_REGEXP.matcher((String) node.getValue());
256                 if (!match.matches()) {
257                     throw new YAMLException("Unexpected timestamp: " + node.getValue());
258                 }
259                 String year_s = match.group(1);
260                 String month_s = match.group(2);
261                 String day_s = match.group(3);
262                 String hour_s = match.group(4);
263                 String min_s = match.group(5);
264                 String sec_s = match.group(6);
265                 String fract_s = match.group(7);
266                 String timezoneh_s = match.group(8);
267                 String timezonem_s = match.group(9);
268 
269                 int usec = 0;
270                 if (fract_s != null) {
271                     usec = Integer.parseInt(fract_s);
272                     if (usec != 0) {
273                         while (10 * usec < 1000) {
274                             usec *= 10;
275                         }
276                     }
277                 }
278                 Calendar cal = Calendar.getInstance();
279                 cal.set(Calendar.YEAR, Integer.parseInt(year_s));
280                 // Java's months are zero-based...
281                 cal.set(Calendar.MONTH, Integer.parseInt(month_s) - 1);
282                 cal.set(Calendar.DAY_OF_MONTH, Integer.parseInt(day_s));
283                 cal.set(Calendar.HOUR_OF_DAY, Integer.parseInt(hour_s));
284                 cal.set(Calendar.MINUTE, Integer.parseInt(min_s));
285                 cal.set(Calendar.SECOND, Integer.parseInt(sec_s));
286                 cal.set(Calendar.MILLISECOND, usec);
287                 if (timezoneh_s != null) {
288                     int zone = 0;
289                     int sign = +1;
290                     if (timezoneh_s.startsWith("-")) {
291                         sign = -1;
292                     }
293                     zone += Integer.parseInt(timezoneh_s.substring(1)) * 3600000;
294                     if (timezonem_s != null) {
295                         zone += Integer.parseInt(timezonem_s) * 60000;
296                     }
297                     cal.set(Calendar.ZONE_OFFSET, sign * zone);
298                 } else {
299                     // no time zone provided
300                     cal.setTimeZone(TimeZone.getTimeZone("UTC"));
301                 }
302                 return cal.getTime();
303             }
304         }

307     private class ConstructYamlOmap extends AbstractConstruct {
308         public Object construct(Node node) {
309             // Note: we do not check for duplicate keys, because it's too
310             // CPU-expensive.
311             Map<Object, Object> omap = new LinkedHashMap<Object, Object>();
312             if (!(node instanceof SequenceNode)) {
313                 throw new ConstructorException("while constructing an ordered map", node
314                         .getStartMark(), "expected a sequence, but found " + node.getNodeId(), node
315                         .getStartMark());
316             }
317             SequenceNode snode = (SequenceNode) node;
318             for (Node subnode : snode.getValue()) {
319                 if (!(subnode instanceof MappingNode)) {
320                     throw new ConstructorException("while constructing an ordered map", node
321                             .getStartMark(), "expected a mapping of length 1, but found "
322                             + subnode.getNodeId(), subnode.getStartMark());
323                 }
324                 MappingNode mnode = (MappingNode) subnode;
325                 if (mnode.getValue().size() != 1) {
326                     throw new ConstructorException("while constructing an ordered map", node
327                             .getStartMark(), "expected a single mapping item, but found "
328                             + mnode.getValue().size() + " items", mnode.getStartMark());
329                 }
330                 Node keyNode = mnode.getValue().get(0)[0];
331                 Node valueNode = mnode.getValue().get(0)[1];
332                 Object key = constructObject(keyNode);
333                 Object value = constructObject(valueNode);
334                 omap.put(key, value);
335             }
336             return omap;
337         }
338     }
