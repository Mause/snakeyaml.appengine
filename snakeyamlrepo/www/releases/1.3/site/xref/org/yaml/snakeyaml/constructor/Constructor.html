33  public class Constructor extends SafeConstructor {
34      private final Map<String, Class<? extends Object>> typeTags;
35      private final Map<Class<? extends Object>, TypeDescription> typeDefinitions;
36  
37      public Constructor() {
38          this(Object.class);
39      }
40  
41      public Constructor(Class<? extends Object> theRoot) {
42          if (theRoot == null) {
43              throw new NullPointerException("Root type must be provided.");
44          }
45          this.yamlConstructors.put(null, new ConstructYamlObject());
46          rootType = theRoot;
47          typeTags = new HashMap<String, Class<? extends Object>>();
48          typeDefinitions = new HashMap<Class<? extends Object>, TypeDescription>();
49      }

51      /***
52       * Create Constructor for a class which does not have to be in the classpath
53       * or for a definition from a Spring ApplicationContext.
54       * 
55       * @param theRoot
56       *            fully qualified class name of the root JavaBean
57       * @throws ClassNotFoundException
58       */
59      public Constructor(String theRoot) throws ClassNotFoundException {
60          this(Class.forName(check(theRoot)));
61      }

63      private static final String check(String s) {
64          if (s == null) {
65              throw new NullPointerException("Root type must be provided.");
66          }
67          if (s.trim().length() == 0) {
68              throw new YAMLException("Root type must be provided.");
69          }
70          return s;
71      }

73      /***
74       * Make YAML aware how to parse a custom Class. If there is no root Class
75       * assigned in constructor then the 'root' property of this definition is
76       * respected.
77       * 
78       * @param definition
79       *            to be added to the Constructor
80       * @return the previous value associated with <tt>definition</tt>, or
81       *         <tt>null</tt> if there was no mapping for <tt>definition</tt>.
82       */
83      public TypeDescription addTypeDescription(TypeDescription definition) {
84          if (definition == null) {
85              throw new NullPointerException("TypeDescription is required.");
86          }
87          if (rootType == Object.class && definition.isRoot()) {
88              rootType = definition.getType();
89          }
90          String tag = definition.getTag();
91          typeTags.put(tag, definition.getType());
92          return typeDefinitions.put(definition.getType(), definition);
93      }

95      private class ConstructYamlObject extends AbstractConstruct {
96          @SuppressWarnings("unchecked")
97          public Object construct(Node node) {
98              Object result = null;
99              try {
100                 Class cl = getClassForNode(node);
101                 java.lang.reflect.Constructor javaConstructor;
102                 switch (node.getNodeId()) {
103                 case mapping:
104                     node.setType(cl);
105                     if (node.isTwoStepsConstruction()) {
106                         result = createJavaBeanNode(node);
107                     } else {
108                         result = constructJavaBeanNode((MappingNode) node);
109                     }
110                     break;
111                 case sequence:
112                     SequenceNode seqNode = (SequenceNode) node;
113                     List<Object> values = (List<Object>) constructSequence(seqNode);
114                     Class[] parameterTypes = new Class[values.size()];
115                     int index = 0;
116                     for (Object parameter : values) {
117                         parameterTypes[index] = parameter.getClass();
118                         index++;
119                     }
120                     javaConstructor = cl.getConstructor(parameterTypes);
121                     Object[] initargs = values.toArray();
122                     result = javaConstructor.newInstance(initargs);
123                     break;
124                 default:// scalar
125                     ScalarNode scaNode = (ScalarNode) node;
126                     Object value = constructScalar(scaNode);
127                     if (Enum.class.isAssignableFrom(cl)) {
128                         String enumValueName = (String) node.getValue();
129                         try {
130                             result = Enum.valueOf(cl, enumValueName);
131                         } catch (Exception ex) {
132                             throw new YAMLException("Unable to find enum value '" + enumValueName
133                                     + "' for enum class: " + cl.getName());
134                         }
135                     } else {
136                         javaConstructor = cl.getConstructor(value.getClass());
137                         result = javaConstructor.newInstance(value);
138                     }
139                 }
140             } catch (Exception e) {
141                 throw new ConstructorException(null, null, "Can't construct a java object for "
142                         + node.getTag() + "; exception=" + e.getMessage(), node.getStartMark(), e);
143             }
144             return result;
145         }
146     }

148     @Override
149     protected Object callConstructor(Node node) {
150         if (Object.class.equals(node.getType()) || "tag:yaml.org,2002:null".equals(node.getTag())) {
151             return super.callConstructor(node);
152         }
153         Object result;
154         switch (node.getNodeId()) {
155         case scalar:
156             result = constructJavaBeanScalarNode((ScalarNode) node);
157             break;
158         case sequence:
159             SequenceNode snode = (SequenceNode) node;
160             if (node.isTwoStepsConstruction()) {
161                 result = createDefaultList(snode.getValue().size());
162             } else {
163                 result = constructSequence(snode);
164             }
165             break;
166         default:// mapping
167             if (Map.class.isAssignableFrom(node.getType())) {
168                 if (node.isTwoStepsConstruction()) {
169                     result = createDefaultMap();
170                 } else {
171                     result = constructMapping((MappingNode) node);
172                 }
173             } else if (Set.class.isAssignableFrom(node.getType())) {
174                 if (node.isTwoStepsConstruction()) {
175                     result = createDefaultSet();
176                 } else {
177                     result = constructSet((MappingNode) node);
178                 }
179             } else {
180                 if (node.isTwoStepsConstruction()) {
181                     result = createJavaBeanNode(node);
182                 } else {
183                     result = constructJavaBeanNode((MappingNode) node);
184                 }
185             }
186         }
187         return result;
188     }

190     @SuppressWarnings("unchecked")
191     @Override
192     protected void callPostCreate(Node node, Object object) {
193         if (!node.isTwoStepsConstruction()) {
194             throw new YAMLException("Inexpected recursive structure. Node: " + node);
195         }
196         if (Object.class.equals(node.getType()) || "tag:yaml.org,2002:null".equals(node.getTag())) {
197             super.callPostCreate(node, object);
198         } else {
199             switch (node.getNodeId()) {
200             case scalar:
201                 throw new YAMLException("Scalars cannot be recursive. Node: " + node);
202             case sequence:
203                 constructSequenceStep2((SequenceNode) node, (List<Object>) object);
204                 break;
205             default:// mapping
206                 if (Map.class.isAssignableFrom(node.getType())) {
207                     constructMapping2ndStep((MappingNode) node, (Map<Object, Object>) object);
208                 } else if (Set.class.isAssignableFrom(node.getType())) {
209                     constructSet2ndStep((MappingNode) node, (Set<Object>) object);
210                 } else {
211                     constructJavaBeanNode2ndStep((MappingNode) node, object);
212                 }
213             }
214         }
215     }

217     private Object constructJavaBeanScalarNode(ScalarNode node) {
218         Class<? extends Object> type = node.getType();
219         Object result;
220         if (type.isPrimitive() || type == String.class || Number.class.isAssignableFrom(type)
221                 || type == Boolean.class || Date.class.isAssignableFrom(type)
222                 || type == Character.class || type == BigInteger.class
223                 || Enum.class.isAssignableFrom(type)) {
224             if (type == String.class) {
225                 Construct stringConstructor = yamlConstructors.get("tag:yaml.org,2002:str");
226                 result = stringConstructor.construct((ScalarNode) node);
227             } else if (type == Boolean.class || type == Boolean.TYPE) {
228                 Construct boolConstructor = yamlConstructors.get("tag:yaml.org,2002:bool");
229                 result = boolConstructor.construct((ScalarNode) node);
230             } else if (type == Character.class || type == Character.TYPE) {
231                 Construct charConstructor = yamlConstructors.get("tag:yaml.org,2002:str");
232                 String ch = (String) charConstructor.construct((ScalarNode) node);
233                 if (ch.length() != 1) {
234                     throw new YAMLException("Invalid node Character: '" + ch + "'; length: "
235                             + ch.length());
236                 }
237                 result = new Character(ch.charAt(0));
238             } else if (Date.class.isAssignableFrom(type)) {
239                 Construct dateConstructor = yamlConstructors.get("tag:yaml.org,2002:timestamp");
240                 Date date = (Date) dateConstructor.construct((ScalarNode) node);
241                 if (type == Date.class) {
242                     result = date;
243                 } else {
244                     try {
245                         java.lang.reflect.Constructor<?> constr = type.getConstructor(long.class);
246                         result = constr.newInstance(date.getTime());
247                     } catch (Exception e) {
248                         throw new YAMLException("Cannot construct: '" + type + "'");
249                     }
250                 }
251             } else if (type == Float.class || type == Double.class || type == Float.TYPE
252                     || type == Double.TYPE || type == BigDecimal.class) {
253                 Construct doubleConstructor = yamlConstructors.get("tag:yaml.org,2002:float");
254                 result = doubleConstructor.construct(node);
255                 if (type == Float.class || type == Float.TYPE) {
256                     result = new Float((Double) result);
257                 } else if (type == BigDecimal.class) {
258                     result = new BigDecimal(((Double) result).doubleValue());
259                 }
260             } else if (type == Byte.class || type == Short.class || type == Integer.class
261                     || type == Long.class || type == BigInteger.class || type == Byte.TYPE
262                     || type == Short.TYPE || type == Integer.TYPE || type == Long.TYPE) {
263                 Construct intConstructor = yamlConstructors.get("tag:yaml.org,2002:int");
264                 result = intConstructor.construct(node);
265                 if (type == Byte.class || type == Byte.TYPE) {
266                     result = new Byte(result.toString());
267                 } else if (type == Short.class || type == Short.TYPE) {
268                     result = new Short(result.toString());
269                 } else if (type == Integer.class || type == Integer.TYPE) {
270                     result = new Integer(result.toString());
271                 } else if (type == Long.class || type == Long.TYPE) {
272                     result = new Long(result.toString());
273                 } else {
274                     // only BigInteger left
275                     result = new BigInteger(result.toString());
276                 }
277             } else if (Enum.class.isAssignableFrom(type)) {
278                 String tag = "tag:yaml.org,2002:" + type.getName();
279                 node.setTag(tag);
280                 result = super.callConstructor(node);
281             } else {
282                 throw new YAMLException("Unsupported class: " + type);
283             }
284         } else {
285             try {
286                 // get value by BaseConstructor
287                 Object value = super.callConstructor(node);
288                 if (type.isArray()) {
289                     result = value;
290                 } else {
291                     java.lang.reflect.Constructor<? extends Object> javaConstructor = type
292                             .getConstructor(value.getClass());
293                     result = javaConstructor.newInstance(value);
294                 }
295             } catch (Exception e) {
296                 throw new YAMLException(e);
297             }
298         }
299         return result;
300     }

302     private Object createJavaBeanNode(Node node) {
303         try {
304             Class<? extends Object> type = node.getType();
305             if (Modifier.isAbstract(type.getModifiers())) {
306                 node.setType(getClassForNode(node));
307             }
308             /***
309              * Using only default constructor. Everything else will be
310              * initialized on 2nd step. If we do here some partial
311              * initialization, how do we then track what need to be done on 2nd
312              * step? I think it is better to get only object here (to have it as
313              * reference for recursion) and do all other thing on 2nd step.
314              */
315             return node.getType().newInstance();
316         } catch (InstantiationException e) {
317             throw new YAMLException(e);
318         } catch (IllegalAccessException e) {
319             throw new YAMLException(e);
320         } catch (ClassNotFoundException e) {
321             throw new YAMLException(e);
322         }
323     }

338     @SuppressWarnings("unchecked")
339     private Object constructJavaBeanNode2ndStep(MappingNode node, Object object) {
340         Class<? extends Object> beanType = node.getType();
341         List<Node[]> nodeValue = (List<Node[]>) node.getValue();
342         for (Node[] tuple : nodeValue) {
343             ScalarNode keyNode;
344             if (tuple[0] instanceof ScalarNode) {
345                 keyNode = (ScalarNode) tuple[0];// key must be scalar
346             } else {
347                 throw new YAMLException("Keys must be scalars but found: " + tuple[0]);
348             }
349             Node valueNode = tuple[1];
350             // keys can only be Strings
351             keyNode.setType(String.class);
352             String key = (String) constructObject(keyNode);
353             boolean isArray = false;
354             try {
355                 Property property = getProperty(beanType, key);
356                 valueNode.setType(property.getType());
357                 TypeDescription memberDescription = typeDefinitions.get(beanType);
358                 if (memberDescription != null) {
359                     switch (valueNode.getNodeId()) {
360                     case sequence:
361                         SequenceNode snode = (SequenceNode) valueNode;
362                         Class<? extends Object> memberType = memberDescription
363                                 .getListPropertyType(key);
364                         if (memberType != null) {
365                             snode.setListType(memberType);
366                         } else if (property.getType().isArray()) {
367                             isArray = true;
368                             snode.setListType(property.getType().getComponentType());
369                         }
370                         break;
371                     case mapping:
372                         MappingNode mnode = (MappingNode) valueNode;
373                         Class<? extends Object> keyType = memberDescription.getMapKeyType(key);
374                         if (keyType != null) {
375                             mnode.setKeyType(keyType);
376                             mnode.setValueType(memberDescription.getMapValueType(key));
377                         }
378                         break;
379                     }
380                 }
381                 Object value = constructObject(valueNode);
382                 if (isArray) {
383                     List<Object> list = (List<Object>) value;
384                     value = list.toArray(createArray(property.getType()));
385                 }
386                 property.set(object, value);
387             } catch (Exception e) {
388                 throw new YAMLException(e);
389             }
390         }
391         return object;
392     }

394     @SuppressWarnings("unchecked")
395     private <T> T[] createArray(Class<T> type) {
396         return (T[]) Array.newInstance(type.getComponentType(), 0);
397     }

399     private Property getProperty(Class<? extends Object> type, String name)
400             throws IntrospectionException {
401         for (PropertyDescriptor property : Introspector.getBeanInfo(type).getPropertyDescriptors()) {
402             if (property.getName().equals(name)) {
403                 if (property.getWriteMethod() != null) {
404                     return new MethodProperty(property);
405                 } else {
406                     throw new YAMLException("Property '" + name + "' on JavaBean: "
407                             + type.getName() + " does not have the write method");
408                 }
409             }
410         }
411         for (Field field : type.getFields()) {
412             int modifiers = field.getModifiers();
413             if (Modifier.isStatic(modifiers) || Modifier.isTransient(modifiers)) {
414                 continue;
415             }
416             if (field.getName().equals(name)) {
417                 return new FieldProperty(field);
418             }
419         }
420         throw new YAMLException("Unable to find property '" + name + "' on class: "
421                 + type.getName());
422     }

424     protected Class<?> getClassForNode(Node node) throws ClassNotFoundException {
425         Class<? extends Object> customTag = typeTags.get(node.getTag());
426         if (customTag == null) {
427             if (node.getTag().length() < "tag:yaml.org,2002:".length()) {
428                 throw new YAMLException("Unknown tag: " + node.getTag());
429             }
430             String name = node.getTag().substring("tag:yaml.org,2002:".length());
431             Class<?> cl = Class.forName(name);
432             typeTags.put(node.getTag(), cl);
433             return cl;
434         } else {
435             return customTag;
436         }
437     }
