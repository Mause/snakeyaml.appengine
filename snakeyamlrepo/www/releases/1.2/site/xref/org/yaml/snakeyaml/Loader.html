22  public class Loader {
23      protected final BaseConstructor constructor;
24      protected Resolver resolver;
25      private boolean attached = false;
26  
27      public Loader(BaseConstructor constructor) {
28          super();
29          this.constructor = constructor;
30      }
31  
32      public Loader() {
33          this(new Constructor());
34      }
35  
36      public Object load(java.io.Reader io) {
37          Composer composer = new Composer(new ParserImpl(new Reader(io)), resolver);
38          constructor.setComposer(composer);
39          return constructor.getSingleData();
40      }
41  
42      public Iterable<Object> loadAll(java.io.Reader yaml) {
43          Composer composer = new Composer(new ParserImpl(new Reader(yaml)), resolver);
44          this.constructor.setComposer(composer);
45          Iterator<Object> result = new Iterator<Object>() {
46              public boolean hasNext() {
47                  return constructor.checkData();
48              }
49  
50              public Object next() {
51                  return constructor.getData();
52              }
53  
54              public void remove() {
55                  throw new UnsupportedOperationException();
56              }
57          };
58          return new YamlIterable(result);
59      }
60  
61      /***
62       * Parse the first YAML document in a stream and produce the corresponding
63       * representation tree.
64       * 
65       * @param yaml
66       *            - YAML document
67       * @return parsed root Node for the specified YAML document
68       */
69      public Node compose(java.io.Reader yaml) {
70          Composer composer = new Composer(new ParserImpl(new Reader(yaml)), resolver);
71          this.constructor.setComposer(composer);
72          return composer.getSingleNode();
73      }
74  
75      /***
76       * Parse all YAML documents in a stream and produce corresponding
77       * representation trees.
78       * 
79       * @param yaml
80       *            - stream of YAML documents
81       * @return parsed root Nodes for all the specified YAML documents
82       */
83      public Iterable<Node> composeAll(java.io.Reader yaml) {
84          final Composer composer = new Composer(new ParserImpl(new Reader(yaml)), resolver);
85          this.constructor.setComposer(composer);
86          Iterator<Node> result = new Iterator<Node>() {
87              public boolean hasNext() {
88                  return composer.checkNode();
89              }
90  
91              public Node next() {
92                  return composer.getNode();
93              }
94  
95              public void remove() {
96                  throw new UnsupportedOperationException();
97              }
98          };
99          return new NodeIterable(result);
100     }

129         this.resolver = resolver;

132     /***
133      * Because Loader is stateful it cannot be shared
134      */
135     void setAttached() {
136         if (!attached) {
137             attached = true;
138         } else {
139             throw new YAMLException("Loader cannot be shared.");
140         }
141     }

143     /***
144      * Parse a YAML stream and produce parsing events.
145      * 
146      * @param yaml
147      *            - YAML document(s)
148      * @return parsed events
149      */
150     public Iterable<Event> parse(java.io.Reader yaml) {
151         final Parser parser = new ParserImpl(new Reader(yaml));
152         Iterator<Event> result = new Iterator<Event>() {
153             public boolean hasNext() {
154                 return parser.peekEvent() != null;
155             }
156 
157             public Event next() {
158                 return parser.getEvent();
159             }
160 
161             public void remove() {
162                 throw new UnsupportedOperationException();
163             }
164         };
165         return new EventIterable(result);
166     }
