71          public static LineBreak getPlatformLineBreak() {
72              String platformLineBreak = System.getProperty("line.separator");
73              for (LineBreak lb : values()) {
74                  if (lb.lineBreak.equals(platformLineBreak)) {
75                      return lb;
76                  }
77              }
78              return LineBreak.UNIX;
79          }
