13      public void testConstructFloat() {
14          Yaml yaml = new Yaml();
15          assertEquals(3.1416, yaml.load("+3.1416"));
16          assertEquals(Double.POSITIVE_INFINITY, yaml.load("+.inf"));
17          assertEquals(Double.POSITIVE_INFINITY, yaml.load(".inf"));
18          assertEquals(Double.NEGATIVE_INFINITY, yaml.load("-.inf"));
19      }

21      public void testSafeConstruct() {

26      public void testSafeConstructJavaBean() {
27          Yaml yaml = new Yaml(new Loader(new SafeConstructor()));
28          String data = "--- !!org.yaml.snakeyaml.constructor.Person\nfirstName: Andrey\nage: 99";
29          try {
30              yaml.load(data);
31              fail("JavaBeans cannot be created by SafeConstructor.");
32          } catch (ConstructorException e) {
33              assertTrue(e
34                      .getMessage()
35                      .contains(
36                              "could not determine a constructor for the tag tag:yaml.org,2002:org.yaml.snakeyaml.constructor.Person"));
37          }
