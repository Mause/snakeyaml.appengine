71      @SuppressWarnings("unchecked")
72      public void testExample_2_22() {
73          YamlDocument document = new YamlDocument("example2_22.yaml");
74          Map<String, Object> map = (Map<String, Object>) document.getNativeData();
75          assertEquals(4, map.size());
76          assertEquals("Expect '2001-12-15T02:59:43.1Z' to be a Date.", Date.class, map.get(
77                  "canonical").getClass());
78          Calendar cal = Calendar.getInstance(TimeZone.getTimeZone("UTC"));
79          cal.clear();
80          cal.set(Calendar.YEAR, 2001);
81          cal.set(Calendar.MONTH, 11); // Java's months are zero-based...
82          cal.set(Calendar.DAY_OF_MONTH, 15);
83          cal.set(Calendar.HOUR_OF_DAY, 2);
84          cal.set(Calendar.MINUTE, 59);
85          cal.set(Calendar.SECOND, 43);
86          cal.set(Calendar.MILLISECOND, 100);
87          Date date = cal.getTime();
88          assertEquals(date, map.get("canonical"));
89          assertEquals("Expect '2001-12-14t21:59:43.10-05:00' to be a Date.", Date.class, map.get(
90                  "iso8601").getClass());
91          assertEquals("Expect '2001-12-14 21:59:43.10 -5' to be a Date.", Date.class, map.get(
92                  "spaced").getClass());
93          assertEquals("Expect '2002-12-14' to be a Date.", Date.class, map.get("date").getClass());
94      }

119     class SomethingConstructor extends Constructor {
120         public SomethingConstructor() {
121             this.yamlConstructors.put("!something", new ConstructSomething());
122         }
123 
124         private class ConstructSomething implements Construct {
125             public Object construct(Node node) {
126                 // convert to upper case
127                 String val = (String) constructScalar((ScalarNode) node);
128                 return val.toUpperCase().replace('\n', ' ').trim();
129             }
130         }
