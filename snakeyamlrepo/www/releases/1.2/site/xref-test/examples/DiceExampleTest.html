33      public void testDiceRepresenter() throws IOException {
34          Dice dice = new Dice(3, 6);
35          Map<String, Dice> data = new HashMap<String, Dice>();
36          data.put("gold", dice);
37          Yaml yaml = new Yaml(new Dumper(new DiceRepresenter(), new DumperOptions()));
38          String output = yaml.dump(data);
39          System.out.println(output);
40          assertEquals("{gold: !dice '3d6'}\n", output);
41      }

43      class DiceRepresenter extends Representer {
44          public DiceRepresenter() {
45              this.representers.put(Dice.class, new RepresentDice());
46          }
47  
48          private class RepresentDice implements Represent {
49              public Node representData(Object data) {
50                  Dice dice = (Dice) data;
51                  String value = dice.getA() + "d" + dice.getB();
52                  return representScalar("!dice", value);
53              }
54          }
55      }
56  
57      class DiceConstructor extends Constructor {
58          public DiceConstructor() {
59              this.yamlConstructors.put("!dice", new ConstructDice());
60          }
61  
62          private class ConstructDice implements Construct {
63              public Object construct(Node node) {
64                  String val = (String) constructScalar((ScalarNode) node);
65                  int position = val.indexOf('d');
66                  Integer a = Integer.parseInt(val.substring(0, position));
67                  Integer b = Integer.parseInt(val.substring(position + 1));
68                  return new Dice(a, b);
69              }
70          }
71      }

73      @SuppressWarnings("unchecked")
74      public void testConstructor() throws IOException {
75          Yaml yaml = new Yaml(new Loader(new DiceConstructor()));
76          Object data = yaml.load("{initial hit points: !dice '8d4'}");
77          Map<String, Dice> map = (Map<String, Dice>) data;
78          assertEquals(new Dice(8, 4), map.get("initial hit points"));

81      @SuppressWarnings("unchecked")
82      public void testImplicitResolver() throws IOException {
83          Yaml yaml = new Yaml(new Loader(new DiceConstructor()), new Dumper(new DiceRepresenter(),
84                  new DumperOptions()));
85          yaml.addImplicitResolver("!dice", Pattern.compile("//d+d//d+"), "123456789");
86          // dump
87          Map<String, Dice> treasure = (Map<String, Dice>) new HashMap<String, Dice>();
88          treasure.put("treasure", new Dice(10, 20));
89          String output = yaml.dump(treasure);
90          System.out.println(output);
91          assertEquals("{treasure: 10d20}\n", output);
92          // load
93          Object data = yaml.load("{damage: 5d10}");
94          Map<String, Dice> map = (Map<String, Dice>) data;
95          assertEquals(new Dice(5, 10), map.get("damage"));
96      }
