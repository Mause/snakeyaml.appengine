12  import org.yaml.snakeyaml.Loader;
13  import org.yaml.snakeyaml.Yaml;
14  import org.yaml.snakeyaml.constructor.Constructor;

16  public class CustomMapExampleTest extends TestCase {
17      @SuppressWarnings("unchecked")
18      public void testMap() throws IOException {
19          Loader loader = new Loader(new CustomConstructor());
20          Yaml yaml = new Yaml(loader);
21          Map data = (Map) yaml.load("{2: '222', 1: '111', 3: '333'}");
22          assertTrue(data instanceof TreeMap);
23          Object[] keys = data.keySet().toArray();
24          // must be sorted
25          assertEquals(new Integer(1), keys[0]);
26          assertEquals(new Integer(2), keys[1]);
27          assertEquals(new Integer(3), keys[2]);
28      }
29  
30      class CustomConstructor extends Constructor {
31          @Override
32          protected Map<Object, Object> createDefaultMap() {
33              return new TreeMap<Object, Object>();
34          }
35      }
